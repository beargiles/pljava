39static jvalue _intArray_coerceDatum(Type self, Datum arg)
40{
41 jvalue result;
42 ArrayType* v = DatumGetArrayTypeP(arg);
43 jsize nElems = (jsize)ArrayGetNItems(ARR_NDIM(v), ARR_DIMS(v));
44 jintArray intArray = JNI_newIntArray(nElems);
45
46 if(ARR_HASNULL(v))
47 {
48 jsize idx;
49 jboolean isCopy = JNI_FALSE;
50 bits8* nullBitMap = ARR_NULLBITMAP(v);
51 jint* values = (jint*)ARR_DATA_PTR(v);
52 jint* elems = JNI_getIntArrayElements(intArray, &isCopy);
53 for(idx = 0; idx < nElems; ++idx)
54 {
55 if(arrayIsNull(nullBitMap, idx))
56 elems[idx] = 0;
57 else
58 elems[idx] = *values++;
59 }
60 JNI_releaseIntArrayElements(intArray, elems, JNI_COMMIT);
61 }
62 else
63 JNI_setIntArrayRegion(intArray, 0, nElems, (jint*)ARR_DATA_PTR(v));
64 result.l = (jobject)intArray;
65 return result;
66}

125 cls = TypeClass_alloc("type.Integer");

129 cls->coerceDatum = _Integer_coerceDatum;
130 cls->coerceObject = _Integer_coerceObject;
131 t_Integer = TypeClass_allocInstance(cls, INT4OID);
132
133 cls = TypeClass_alloc("type.int");
134 cls->JNISignature = "I";
135 cls->javaTypeName = "int";
136 cls->invoke = _int_invoke;
137 cls->coerceDatum = _int_coerceDatum;

139 cls->createArrayType = _int_createArrayType;
140 s_intClass = cls;
141
142 t_int = TypeClass_allocInstance(cls, INT4OID);
143 t_int->objectType = t_Integer;
144 Type_registerType("int", t_int);
145 Type_registerType("java.lang.Integer", t_Integer);
146}
