22 * byte[] type. Copies data to/from a bytea struct.
23 */
24static jvalue _byte_array_coerceDatum(Type self, Datum arg)
25{
26 jvalue result;
27 bytea* bytes = DatumGetByteaP(arg);
28 jsize length = VARSIZE(bytes) - VARHDRSZ;
29 jbyteArray ba = JNI_newByteArray(length);
30 JNI_setByteArrayRegion(ba, 0, length, (jbyte*)VARDATA(bytes));
31 result.l = ba;
32 return result;
33}

35static Datum _byte_array_coerceObject(Type self, jobject byteArray)
36{
37 bytea* bytes = 0;
38 if(byteArray == 0)
39 return 0;
40
41 if(JNI_isInstanceOf(byteArray, s_byteArray_class))
42 {
43 jsize length = JNI_getArrayLength((jarray)byteArray);
44 int32 byteaSize = length + VARHDRSZ;
45
46 bytes = (bytea*)palloc(byteaSize);
47 SET_VARSIZE(bytes, byteaSize);
48 JNI_getByteArrayRegion((jbyteArray)byteArray, 0, length, (jbyte*)VARDATA(bytes));
49 }
50 else if(JNI_isInstanceOf(byteArray, s_BlobValue_class))
51 {
52 jobject byteBuffer;
53 int32 byteaSize;
54 jlong length = JNI_callLongMethod(byteArray, s_BlobValue_length);
55
56 byteaSize = (int32)(length + VARHDRSZ);
57 bytes = (bytea*)palloc(byteaSize);
58 SET_VARSIZE(bytes, byteaSize);
59
60 byteBuffer = JNI_newDirectByteBuffer((void*)VARDATA(bytes), length);
61 if(byteBuffer != 0)
62 JNI_callVoidMethod(byteArray, s_BlobValue_getContents, byteBuffer);
63 JNI_deleteLocalRef(byteBuffer);
64 }
65 else
66 {
67 Exception_throwIllegalArgument("Not coercable to bytea");
68 }
69
70 PG_RETURN_BYTEA_P(bytes);
71}
