29jobject TupleTable_createFromSlot(TupleTableSlot* tts)
30{
31 HeapTuple tuple;
32 jobject tupdesc;
33 jobjectArray tuples;
34 MemoryContext curr;
35
36 if(tts == 0)
37 return 0;
38
39 curr = MemoryContextSwitchTo(JavaMemoryContext);
40
41 tupdesc = pljava_TupleDesc_internalCreate(tts->tts_tupleDescriptor);
42 tuple = ExecCopySlotHeapTuple(tts);
43 tuples = pljava_Tuple_createArray(&tuple, 1, false);
44
45 MemoryContextSwitchTo(curr);
46
47 return JNI_newObject(s_TupleTable_class, s_TupleTable_init, tupdesc, tuples);
48}

50jobject TupleTable_create(SPITupleTable* tts, jobject knownTD)
51{
52 jobjectArray tuples;
53 uint64 tupcount;
54 MemoryContext curr;
55
56 if(tts == 0)
57 return 0;
58
59#if PG_VERSION_NUM < 130000
60 tupcount = tts->alloced - tts->free;
61#else
62 tupcount = tts->numvals;
63#endif
64 if ( tupcount > PG_INT32_MAX )
65 ereport(ERROR,
66 (errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
67 errmsg("a PL/Java TupleTable cannot represent more than "
68 "INT32_MAX rows")));
69
70 curr = MemoryContextSwitchTo(JavaMemoryContext);
71
72 if(knownTD == 0)
73 knownTD = pljava_TupleDesc_internalCreate(tts->tupdesc);
74
75 tuples = pljava_Tuple_createArray(tts->vals, (jint)tupcount, true);
76 MemoryContextSwitchTo(curr);
77
78 return JNI_newObject(s_TupleTable_class, s_TupleTable_init, knownTD, tuples);
79}
