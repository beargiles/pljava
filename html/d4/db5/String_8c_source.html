72Datum _String_coerceObject(Type self, jobject jstr)
73{
74 char* tmp;
75 Datum ret;
76 if(jstr == 0)
77 return 0;
78
79 jstr = JNI_callObjectMethod(jstr, s_Object_toString);
80 if(JNI_exceptionCheck())
81 return 0;
82
83 tmp = String_createNTS(jstr);
84 JNI_deleteLocalRef(jstr);
85
86 ret = FunctionCall3(
87 &((PLJString)self)->textInput,
88 CStringGetDatum(tmp),
89 ObjectIdGetDatum(((PLJString)self)->elementType),
90 Int32GetDatum(-1));
91 pfree(tmp);
92 return ret;

118jstring String_createJavaString(text* t)
119{
120 jstring result = 0;
121 if(t != 0)
122 {
123 jobject bytebuf;
124 jobject charbuf;
125 char* src = VARDATA(t);
126 char* utf8 = src;
127 Size srcLen = VARSIZE(t) - VARHDRSZ;
128 if(srcLen == 0)
129 return s_the_empty_string;
130
131 if ( s_two_step_conversion )
132 {
133 utf8 = (char*)pg_do_encoding_conversion((unsigned char*)src,
134 (int)srcLen, s_server_encoding, PG_UTF8);
135 /* pg_do_encoding_conversion may return the source argument
136 * unchanged in more circumstances than you'd expect. As the source
137 * argument isn't NUL-terminated, don't call strlen on it.
138 */
139 if (utf8 != src)
140 srcLen = strlen(utf8);
141 }
142 bytebuf = JNI_newDirectByteBuffer(utf8, srcLen);
143 charbuf = JNI_callObjectMethodLocked(s_CharsetDecoder_instance,
144 s_CharsetDecoder_decode, bytebuf);
145 result = JNI_callObjectMethodLocked(charbuf, s_Object_toString);
146
147 JNI_deleteLocalRef(bytebuf);
148 JNI_deleteLocalRef(charbuf);
149 /* pg_do_encoding_conversion will return the source argument
150 * when no conversion is required. We don't want to accidentally
151 * free that pointer.
152 */
153 if(utf8 != src)
154 pfree(utf8);
155 }
156 return result;
157}

160{
161 jstring result = 0;
162 if(cp != 0)
163 {
164 jobject bytebuf;
165 jobject charbuf;
166 Size sz = strlen(cp);
167 char const * utf8 = cp;
168 if ( s_two_step_conversion )
169 {
170 utf8 = (char*)pg_do_encoding_conversion((unsigned char*)cp,
171 (int)sz, s_server_encoding, PG_UTF8);
172 /* Here the source is NUL-terminated, so calling strlen on it
173 * would be safe, but unnecessary all the same.
174 */
175 if ( utf8 != cp )
176 sz = strlen(utf8);
177 }
178 bytebuf = JNI_newDirectByteBuffer((void *)utf8, sz);
179 charbuf = JNI_callObjectMethodLocked(s_CharsetDecoder_instance,
180 s_CharsetDecoder_decode, bytebuf);
181 result = JNI_callObjectMethodLocked(charbuf, s_Object_toString);

183 JNI_deleteLocalRef(bytebuf);
184 JNI_deleteLocalRef(charbuf);
185 /* pg_do_encoding_conversion will return the source argument
186 * when no conversion is required. We don't want to accidentally
187 * free that pointer.
188 */
189 if(utf8 != cp)
190 pfree((void *)utf8);
191 }
192 return result;

197text* String_createText(jstring javaString)
198{
199 text* result = 0;
200 if(javaString != 0)
201 {
202 char* denc;
203 Size dencLen;
204 Size varSize;
205 jobject charbuf = JNI_callStaticObjectMethodLocked(s_CharBuffer_class,
206 s_CharBuffer_wrap, javaString);
207 StringInfoData sid;
208 initStringInfo(&sid);
209 appendCharBuffer(&sid, charbuf);
210 JNI_deleteLocalRef(charbuf);
211 denc = sid.data;
212 dencLen = sid.len;
213 if ( s_two_step_conversion )
214 {
215 denc = (char*)pg_do_encoding_conversion(
216 (unsigned char*)denc, (int)dencLen, PG_UTF8, s_server_encoding);
217 /* pg_do_encoding_conversion may return the source argument
218 * unchanged in more circumstances than you'd expect. As the source
219 * argument isn't NUL-terminated, don't call strlen on it.
220 */
221 if (denc != sid.data)
222 dencLen = strlen(denc);
223 }
224 varSize = dencLen + VARHDRSZ;
225
226 /* Allocate and initialize the text structure.
227 */
228 result = (text*)palloc(varSize);
229 SET_VARSIZE(result, varSize); /* Total size of structure, not just data */
230 memcpy(VARDATA(result), denc, dencLen);
231
232 if(denc != sid.data)
233 pfree(denc);
234 pfree(sid.data);
235 }
236 return result;
237}

239char* String_createNTS(jstring javaString)
240{
241 char* result = 0;
242
243 if( 0 == javaString )
244 return result;
245
246 if ( uninitialized )
247 {
248 const char* u8buf;
249
250 s_server_encoding = GetDatabaseEncoding();
251 u8buf = JNI_getStringUTFChars( javaString, NULL);
252 if ( 0 == u8buf )
253 return result;
254 result = (char*)pg_do_encoding_conversion(
255 (unsigned char *)u8buf, (int)strlen( u8buf),
256 PG_UTF8, s_server_encoding);
257 if ( result == u8buf )
258 result = pstrdup( result);
259 JNI_releaseStringUTFChars( javaString, u8buf);
260 return result;
261 }
262 else
263 {
264 jobject charbuf = JNI_callStaticObjectMethodLocked(s_CharBuffer_class,
265 s_CharBuffer_wrap, javaString);
266 StringInfoData sid;
267 initStringInfo(&sid);
268 appendCharBuffer(&sid, charbuf);
269 JNI_deleteLocalRef(charbuf);
270
271 result = (char*)pg_do_encoding_conversion(
272 (unsigned char *)sid.data, sid.len, PG_UTF8, s_server_encoding);
273
274 /* pg_do_encoding_conversion will return the source argument
275 * when no conversion is required. Don't free it in that case.
276 */
277 if(result != sid.data)
278 pfree(sid.data);
279 }
280
281 return result;
282}

284void String_appendJavaString(StringInfoData* buf, jstring javaString)
285{
286 if ( 0 == javaString )
287 return;
288 if ( ! s_two_step_conversion )
289 {
290 jobject charbuf = JNI_callStaticObjectMethodLocked(s_CharBuffer_class,
291 s_CharBuffer_wrap, javaString);
292 appendCharBuffer(buf, charbuf);
293 JNI_deleteLocalRef(charbuf);
294 }
295 else
296 {
297 char* dbEnc = String_createNTS(javaString);
298 if ( 0 == dbEnc ) /* this can happen if a JNI call fails */
299 return;
300 appendStringInfoString(buf, dbEnc);
301 pfree(dbEnc);
302 }
303}

305static void appendCharBuffer(StringInfoData* buf, jobject charbuf)
306{
307 Size nchars;
308 char *bp;
309 Size cap;
310 jobject bytebuf;
311 jobject coderresult;
312
313 for ( ;; )
314 {
315 /*
316 * Invariant: charbuf has some chars to encode, buf _might_ have room.
317 * Broken StringInfo invariant: within this loop, might lack end NUL.
318 */
319 nchars = JNI_callIntMethodLocked(charbuf, s_Buffer_remaining);
320 /*
321 * enlargeStringInfo does nothing if it's already large enough, and
322 * enlarges generously if it isn't, not by nickels and dimes.
323 */
324 cap = (Size)(s_CharsetEncoder_averageBytesPerChar * (double)nchars);
325 enlargeStringInfo(buf, (int)cap);
326 /*
327 * Give the JVM a window into the unused portion of buf.
328 */
329 bp = buf->data + buf->len;
330 cap = buf->maxlen - buf->len;
331 bytebuf = JNI_newDirectByteBuffer(bp, cap);
332 /*
333 * Encode as much as will fit, then update StringInfo len to reflect it.
334 */
335 coderresult = JNI_callObjectMethodLocked(s_CharsetEncoder_instance,
336 s_CharsetEncoder_encode, charbuf, bytebuf, (jboolean)JNI_TRUE);
337 buf->len += JNI_callIntMethodLocked(bytebuf, s_Buffer_position);
338 JNI_deleteLocalRef(bytebuf);
339
340 if ( ! JNI_isSameObject(coderresult, s_CoderResult_OVERFLOW) )
341 break;
342 JNI_deleteLocalRef(coderresult);
343 }
344 /*
345 * Remember the StringInfo-is-NUL-terminated invariant might not hold here.
346 */
347 if ( JNI_isSameObject(coderresult, s_CoderResult_UNDERFLOW) )
348 if ( 0 == JNI_callIntMethodLocked(charbuf, s_Buffer_remaining) )
349 {
350 JNI_deleteLocalRef(coderresult);
351 enlargeStringInfo(buf, 1); /* MOST PROBABLY a no-op */
352 buf->data[buf->len] = '\0'; /* I want my invariant back! */
353 return;
354 }
355 JNI_callVoidMethodLocked(coderresult, s_CoderResult_throwException);
356}

360void String_initialize(void)
361{
362 s_Object_class = (jclass)JNI_newGlobalRef(PgObject_getJavaClass("java/lang/Object"));
363 s_Object_toString = PgObject_getJavaMethod(s_Object_class, "toString", "()Ljava/lang/String;");
364 s_String_class = (jclass)JNI_newGlobalRef(PgObject_getJavaClass("java/lang/String"));
365
366 s_StringClass = TypeClass_alloc2("type.String", sizeof(struct TypeClass_), sizeof(struct String_));
367 s_StringClass->JNISignature = "Ljava/lang/String;";
368 s_StringClass->javaTypeName = "java.lang.String";
369 s_StringClass->canReplaceType = _String_canReplaceType;
370 s_StringClass->coerceDatum = _String_coerceDatum;
371 s_StringClass->coerceObject = _String_coerceObject;
372
373 /*
374 * Frame push/pop hoisted here out of String_initialize_codec to mollify
375 * pre-C99 compilers that don't want that function to have declarations
376 * after a statement.
377 */
378 JNI_pushLocalFrame(16);
379 String_initialize_codec();
380 JNI_popLocalFrame(NULL);
381
382 /*
383 * Registering known types will increase the performance
384 * a bit. The "default" is used when all else fails.
385 */
386 Type_registerType2(TEXTOID, 0, String_obtain);
387 Type_registerType2(CSTRINGOID, 0, String_obtain);
388 Type_registerType2(BPCHAROID, 0, String_obtain);
389 Type_registerType2(NAMEOID, 0, String_obtain);
390 Type_registerType2(VARCHAROID, "java.lang.String", String_obtain);
391}

393static void String_initialize_codec()
394{
395 /*
396 * Wondering why this function doesn't bother deleting its many local refs?
397 * The call is wrapped in pushLocalFrame/popLocalFrame in the caller.
398 */
399 jmethodID string_intern = PgObject_getJavaMethod(s_String_class,
400 "intern", "()Ljava/lang/String;");
401 jstring empty = JNI_newStringUTF( "");
402 jclass charset_class =
403 PgObject_getJavaClass("java/nio/charset/Charset");
404 jmethodID charset_newDecoder = PgObject_getJavaMethod(charset_class,
405 "newDecoder", "()Ljava/nio/charset/CharsetDecoder;");
406 jmethodID charset_newEncoder = PgObject_getJavaMethod(charset_class,
407 "newEncoder", "()Ljava/nio/charset/CharsetEncoder;");
408 jclass decoder_class =
409 PgObject_getJavaClass("java/nio/charset/CharsetDecoder");
410 jclass encoder_class =
411 PgObject_getJavaClass("java/nio/charset/CharsetEncoder");
412 jmethodID encoder_abpc =
413 PgObject_getJavaMethod(encoder_class, "averageBytesPerChar", "()F");
414 jclass result_class = PgObject_getJavaClass("java/nio/charset/CoderResult");
415 jfieldID overflow = PgObject_getStaticJavaField(result_class, "OVERFLOW",
416 "Ljava/nio/charset/CoderResult;");
417 jfieldID underflow = PgObject_getStaticJavaField(result_class, "UNDERFLOW",
418 "Ljava/nio/charset/CoderResult;");
419 jclass buffer_class = PgObject_getJavaClass("java/nio/Buffer");
420 jobject servercs;
421
422 /*
423 * Records what the final state of s_two_step_conversion will be, but the
424 * static is left at its initial value until all preparations are complete.
425 */
426 bool two_step_when_ready = s_two_step_conversion;
427
428 s_server_encoding = GetDatabaseEncoding();
429
430 if ( PG_SQL_ASCII == s_server_encoding )
431 {
432 jmethodID forname =
433 PgObject_getStaticJavaMethod(charset_class,
434 "forName", "(Ljava/lang/String;)Ljava/nio/charset/Charset;");
435 jstring sql_ascii = JNI_newStringUTF("X-PGSQL_ASCII");
436
437 two_step_when_ready = false;
438
439 servercs = JNI_callStaticObjectMethodLocked(charset_class,
440 forname, sql_ascii);
441 }
442 else
443 {
444 jclass scharset_class =
445 PgObject_getJavaClass("java/nio/charset/StandardCharsets");
446 jfieldID scharset_UTF_8 = PgObject_getStaticJavaField(scharset_class,
447 "UTF_8", "Ljava/nio/charset/Charset;");
448
449 two_step_when_ready = PG_UTF8 != s_server_encoding;
450
451 servercs = JNI_getStaticObjectField(scharset_class, scharset_UTF_8);
452 }
453
454 s_CharsetDecoder_instance =
455 JNI_newGlobalRef(JNI_callObjectMethod(servercs, charset_newDecoder));
456 s_CharsetEncoder_instance =
457 JNI_newGlobalRef(JNI_callObjectMethod(servercs, charset_newEncoder));
458 s_CharsetDecoder_decode = PgObject_getJavaMethod(decoder_class, "decode",
459 "(Ljava/nio/ByteBuffer;)Ljava/nio/CharBuffer;");
460 s_CharsetEncoder_encode = PgObject_getJavaMethod(encoder_class, "encode",
461 "(Ljava/nio/CharBuffer;Ljava/nio/ByteBuffer;Z)"
462 "Ljava/nio/charset/CoderResult;");
463 s_CharsetEncoder_averageBytesPerChar =
464 JNI_callFloatMethod(s_CharsetEncoder_instance, encoder_abpc);
465 s_CoderResult_OVERFLOW = JNI_newGlobalRef(
466 JNI_getStaticObjectField(result_class, overflow));
467 s_CoderResult_UNDERFLOW = JNI_newGlobalRef(
468 JNI_getStaticObjectField(result_class, underflow));
469 s_CoderResult_throwException = PgObject_getJavaMethod(result_class,
470 "throwException", "()V");
471 s_CharBuffer_class = (jclass)JNI_newGlobalRef(
472 PgObject_getJavaClass("java/nio/CharBuffer"));
473 s_CharBuffer_wrap = PgObject_getStaticJavaMethod(s_CharBuffer_class,
474 "wrap", "(Ljava/lang/CharSequence;)Ljava/nio/CharBuffer;");
475 s_Buffer_position = PgObject_getJavaMethod(buffer_class,
476 "position", "()I");
477 s_Buffer_remaining = PgObject_getJavaMethod(buffer_class,
478 "remaining", "()I");
479
480 s_the_empty_string = JNI_newGlobalRef(
481 JNI_callObjectMethod(empty, string_intern));
482
483 s_two_step_conversion = two_step_when_ready;
484 uninitialized = false;
485}
