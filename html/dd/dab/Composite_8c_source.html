71/*
72 * This function is a bit special in that it adds an additional parameter
73 * to the parameter list (a java.sql.ResultSet implemented as a
74 * SingleRowWriter) and calls a boolean method. It's assumed that the
75 * SingleRowWriter has been initialized with values if the method returns
76 * true. If so, the values are obtained in the form of a HeapTuple which in
77 * turn is returned (as a Datum) from this method.
78 */
79static Datum _Composite_invoke(Type self, Function fn, PG_FUNCTION_ARGS)
80{
81 bool hasRow;
82 Datum result = 0;
83 TupleDesc tupleDesc = Type_getTupleDesc(self, fcinfo);
84 jobject jtd = pljava_TupleDesc_create(tupleDesc);
85 jvalue singleRowWriter;
86 singleRowWriter.l = _createWriter(jtd);
87 /*
88 * Caller guarantees room for one extra reference parameter, so it will go
89 * at index (length - 1).
90 */
91 pljava_Function_setParameter(fn, -1, singleRowWriter);
92
93 hasRow = (pljava_Function_booleanInvoke(fn) == JNI_TRUE);
94
95 if(hasRow)
96 {
97 /* Obtain tuple and return it as a Datum. Must be done using a more
98 * durable context.
99 */
100 MemoryContext currCtx = Invocation_switchToUpperContext();
101 HeapTuple tuple = _getTupleAndClear(singleRowWriter.l);
102 result = HeapTupleGetDatum(tuple);
103 MemoryContextSwitchTo(currCtx);
104 }
105 else
106 fcinfo->isnull = true;
107
108 JNI_deleteLocalRef(jtd);
109 JNI_deleteLocalRef(singleRowWriter.l);
110 return result;
111}

161static TupleDesc _Composite_getTupleDesc(Type self, PG_FUNCTION_ARGS)
162{
163 TupleDesc td = ((Composite)self)->m_tupleDesc;
164 if(td != 0)
165 return td;
166
167 switch(get_call_result_type(fcinfo, 0, &td))
168 {
169 case TYPEFUNC_COMPOSITE:
170 case TYPEFUNC_RECORD:
171 if(td == 0)
172 {
173 ereport(ERROR,
174 (errcode(ERRCODE_DATATYPE_MISMATCH),
175 errmsg("function returning record is missing "
176 " result type (try AS?)")));
177 break;
178 }
179 if(td->tdtypeid == RECORDOID)
180 /*
181 * We can't hold on to this one. It's anonymous
182 * and may vary between calls.
183 */
184 td = CreateTupleDescCopy(td);
185 else
186 {
187 td = createGlobalTupleDescCopy(td);
188 ((Composite)self)->m_tupleDesc = td;
189 }
190 break;
191 default:
192 ereport(ERROR,
193 (errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
194 errmsg("function returning record called in context "
195 "that cannot accept type record")));
196 }
197 return td;
198}

200Type Composite_obtain(Oid typeId)
201{
202 Composite infant = (Composite)TypeClass_allocInstance(s_CompositeClass, typeId);
203 if(typeId == RECORDOID)
204 infant->m_tupleDesc = 0;
205 else
206 {
207 TupleDesc tmp = lookup_rowtype_tupdesc(typeId, -1);
208 infant->m_tupleDesc = createGlobalTupleDescCopy(tmp);
209 ReleaseTupleDesc(tmp);
210 }
211 return (Type)infant;
212}
