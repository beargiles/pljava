39static jvalue _shortArray_coerceDatum(Type self, Datum arg)
40{
41 jvalue result;
42 ArrayType* v = DatumGetArrayTypeP(arg);
43 jsize nElems = (jsize)ArrayGetNItems(ARR_NDIM(v), ARR_DIMS(v));
44 jshortArray shortArray = JNI_newShortArray(nElems);
45
46 if(ARR_HASNULL(v))
47 {
48 jsize idx;
49 jboolean isCopy = JNI_FALSE;
50 bits8* nullBitMap = ARR_NULLBITMAP(v);
51 jshort* values = (jshort*)ARR_DATA_PTR(v);
52 jshort* elems = JNI_getShortArrayElements(shortArray, &isCopy);
53 for(idx = 0; idx < nElems; ++idx)
54 {
55 if(arrayIsNull(nullBitMap, idx))
56 elems[idx] = 0;
57 else
58 elems[idx] = *values++;
59 }
60 JNI_releaseShortArrayElements(shortArray, elems, JNI_COMMIT);
61 }
62 else
63 JNI_setShortArrayRegion(shortArray, 0, nElems, (jshort*)ARR_DATA_PTR(v));
64 result.l = (jobject)shortArray;
65 return result;
66}
