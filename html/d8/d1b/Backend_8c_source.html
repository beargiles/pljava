71/* About PGDLLEXPORT. It didn't exist before PG 9.0. In that revision it was
72 * defined (for Windows) as __declspec(dllexport) for MSVC, but as
73 * __declspec(dllimport) for any other toolchain. That was quickly changed
74 * (for 9.0.2 and ever since) as still __declspec(dllexport) for MSVC, but
75 * empty for any other toolchain. The explanation for that (in PG commit
76 * 844ed5d in November 2010) was that "dllexport and dllwrap don't work well
77 * together." There are records as far back as 2002 anyway
78 * (e.g. http://lists.gnu.org/archive/html/libtool/2002-09/msg00069.html)
79 * calling dllwrap deprecated, PL/Java's Maven build certainly doesn't
80 * use it, and I don't know what it would do if it did. It seems too brittle
81 * to rely on whatever PGDLLEXPORT might happen to mean across PG versions,
82 * and wiser for the moment to cleanly define something here, for the all of
83 * three(*) symbols that need it.
84 *
85 * The only case where it expands to anything is when building with Microsoft
86 * Visual Studio. When building with other toolchains it just goes away, even
87 * on Windows when building with MinGW (the only other Windows toolchain
88 * tested). MinGW can work either way: selectively exporting things based on
89 * a __declspec, or with the --export-all-symbols linker option so everything
90 * is visible, as on a *n*x platform. PL/Java could in theory choose either
91 * approach, but for one detail: there is a (*)fourth symbol that needs to be
92 * exported. PG_MODULE_MAGIC defines one, and being a PostgreSQL-supplied macro,
93 * it uses PGDLLEXPORT, which expands to nothing for MinGW (in recent PG
94 * versions anyway), forcing --export-all-symbols as the answer for MinGW.
95 */
96#ifdef _MSC_VER
97#define PLJAVADLLEXPORT __declspec (dllexport)
98#else
99#define PLJAVADLLEXPORT
100#endif

235/*
236 * In a background worker, _PG_init may be called very early, before much of
237 * the state needed during PL/Java initialization has even been set up. When
238 * that case is detected, initsequencer needs to go just as far as
239 * IS_GUCS_REGISTERED and then bail. The GUC assign hooks may then also be
240 * invoked as GUC values get copied from the lead process; they also need to
241 * return quickly (accomplished by checking this flag in ASSIGNRETURNIFNXACT).
242 * Further initialization is thus deferred until the first actual call arrives
243 * at the call handler, which resets this flag and rejoins the initsequencer.
244 * The same lazy approach needs to be followed during a pg_upgrade (which test-
245 * loads libraries, thus calling _PG_init). This flag is set for either case.
246 */

535/*
536 * There are a few ways to arrive in the initsequencer.
537 * 1. From _PG_init (called exactly once when the library is loaded for ANY
538 * reason).
539 * 1a. Because of the command LOAD 'libraryname';
540 * This case can be distinguished because _PG_init will have found the
541 * LOAD command and saved the 'libraryname' in pljavaLoadPath.
542 * 1b. Because of a CREATE FUNCTION naming this library. pljavaLoadPath will
543 * be NULL.
544 * 1c. By the first actual use of a PL/Java function, causing this library
545 * to be loaded. pljavaLoadPath will be NULL. The called function's Oid
546 * will be available to the call handler once we return from _PG_init,
547 * but it isn't (easily) available here.
548 * 2. From the call handler, if initialization isn't complete yet. That can only
549 * mean something failed in the earlier call to _PG_init, and whatever it was
550 * is highly likely to fail again. That may lead to the untidyness of
551 * duplicated diagnostic messages, but for now I like the belt-and-suspenders
552 * approach of making sure the init sequence gets as many chances as possible
553 * to succeed.
554 * 3. From a GUC assign hook, if the user has updated a setting that might allow
555 * initialization to succeed. It resumes from where it left off.
556 * 4. From the validator handler, if initialization isn't complete yet. That
557 * will definitely happen during pg_upgrade, which is a case where deferInit
558 * will have been set. The validator will then clear deferInit and try to get
559 * further in the init sequence. Importantly, pg_upgrade also sets
560 * check_function_bodies false, which limits the validator's work to a syntax
561 * check of the AS string. The validator therefore will not need to obtain a
562 * schemaLoader or do anything else that requires the sqlj schema to be fully
563 * populated (as, during pg_upgrade, it may not yet be). However, the
564 * validator handler must avoid any action that sets pljavaLoadPath, as a
565 * non-NULL value there would be treated below as case 1a, and trigger an
566 * attempt to set up the sqlj schema.
567 *
568 * In all cases, the sequence must progress as far as starting the VM and
569 * initializing the PL/Java classes. In all cases except 1a, that's enough,
570 * assuming the language handlers and schema have all been set up already (or,
571 * in case 1b, the user is intent on setting them up explicitly).
572 *
573 * In case 1a, we can go ahead and test for, and create, the schema, functions,
574 * and language entries as needed, using pljavaLoadPath as the library path
575 * if creating the language handler functions. One-stop shopping. (The presence
576 * of pljavaLoadPath in any of the other cases, such as resumption by an assign
577 * hook, indicates it is really a continuation of case 1a.)
578 */
579static void initsequencer(enum initstage is, bool tolerant)
580{
581 JVMOptList optList;
582 Invocation ctx;
583 jint JNIresult;
584 char *greeting;
585
586 switch (is)
587 {
588 case IS_FORMLESS_VOID:
589 registerGUCOptions();
590 initstage = IS_GUCS_REGISTERED;
591 if ( deferInit )
592 return;
593 warnJEP411 = false;
594 /*FALLTHROUGH*/
595
596 case IS_GUCS_REGISTERED:
597 if ( NULL == libjvmlocation )
598 {
599 ereport(WARNING, (
600 errmsg("Java virtual machine not yet loaded"),
601 errdetail("location of libjvm is not configured"),
602 errhint("SET pljava.libjvm_location TO the correct "
603 "path to the jvm library (libjvm.so or jvm.dll, etc.)")));
604 goto check_tolerant;
605 }
606 initstage = IS_CAND_JVMLOCATION;
607 /*FALLTHROUGH*/
608
609 case IS_CAND_JVMLOCATION:
610 if ( NULL == policy_urls )
611 {
612 ereport(WARNING, (
613 errmsg("Java virtual machine not yet loaded"),
614 errdetail("Java policy URL(s) not configured"),
615 errhint("SET pljava.policy_urls TO the security policy "
616 "files PL/Java is to use.")));
617 goto check_tolerant;
618 }
619 initstage = IS_CAND_POLICYURLS;
620 /*FALLTHROUGH*/
621
622 case IS_CAND_POLICYURLS:
623 if ( ! pljavaEnabled )
624 {
625 ereport(WARNING, (
626 errmsg("Java virtual machine not yet loaded"),
627 errdetail(
628 "Pausing because \"pljava.enable\" is set \"off\". "),
629 errhint(
630 "After changing any other settings as necessary, set it "
631 "\"on\" to proceed.")));
632 goto check_tolerant;
633 }
634 initstage = IS_PLJAVA_ENABLED;
635 /*FALLTHROUGH*/
636
637 case IS_PLJAVA_ENABLED:
638 libjvm_handle = pg_dlopen(libjvmlocation);
639 if ( NULL == libjvm_handle )
640 {
641 ereport(WARNING, (
642 errmsg("Java virtual machine not yet loaded"),
643 errdetail("%s", (char *)pg_dlerror()),
644 errhint("SET pljava.libjvm_location TO the correct "
645 "path to the jvm library (libjvm.so or jvm.dll, etc.)")));
646 goto check_tolerant;
647 }
648 initstage = IS_CAND_JVMOPENED;
649 /*FALLTHROUGH*/
650
651 case IS_CAND_JVMOPENED:
652 pljava_createvm =

917 * A function having everything to do with logging, which ought to be factored

920static void reLogWithChangedLevel(int level)
921{
922 ErrorData *edata = CopyErrorData();
923 int sqlstate = edata->sqlerrcode;
924 int category = ERRCODE_TO_CATEGORY(sqlstate);
925 FlushErrorState();
926 if ( WARNING > level )
927 {
928 if ( ERRCODE_SUCCESSFUL_COMPLETION != category )
929 sqlstate = ERRCODE_SUCCESSFUL_COMPLETION;
930 }
931 else if ( WARNING == level )
932 {
933 if ( ERRCODE_WARNING != category && ERRCODE_NO_DATA != category )
934 sqlstate = ERRCODE_WARNING;
935 }
936 else if ( ERRCODE_WARNING == category || ERRCODE_NO_DATA == category ||
937 ERRCODE_SUCCESSFUL_COMPLETION == category )
938 sqlstate = ERRCODE_INTERNAL_ERROR;
939
940 edata->elevel = level;
941 edata->sqlerrcode = sqlstate;
942 PG_TRY();
943 {
944 ThrowErrorData(edata);
945 }
946 PG_CATCH();
947 {
948 FreeErrorData(edata); /* otherwise this wouldn't happen in ERROR case */
949 PG_RE_THROW();
950 }
951 PG_END_TRY();
952 FreeErrorData(edata);
953}

1171static jint JNICALL my_vfprintf(FILE* fp, const char* format, va_list args)
1172{
1173 static char const * const cap_format =
1174 "WARNING: JNI local refs: %u, exceeds capacity: %u";
1175 static char const at_prefix[] = "\tat ";
1176 static char const locked_prefix[] = "\t- locked <";
1177 static char const class_prefix[] = "(a ";
1178 static char const culprit[] =
1179 " com.sun.management.internal.DiagnosticCommandImpl.";
1180 static char const nostack[] =
1181 "No stacktrace, probably called from PostgreSQL";
1182 static enum matchstate
1183 {
1184 VFP_INITIAL,
1185 VFP_MAYBE,
1186 VFP_ATE_AT,
1187 VFP_ATE_LOCKED
1188 }
1189 state = VFP_INITIAL;
1190 static unsigned int lastlive, lastcap;
1191
1192 char buf[1024];
1193 char* ep;
1194 char* bp = buf;
1195 unsigned int live, cap;
1196 int got;
1197 char const *detail;
1198
1199 vsnprintf(buf, sizeof(buf), format, args);
1200
1201 /* Try to eliminate annoying -Xcheck:jni messages from deep in JMX that
1202 * nothing can be done about here.
1203 */
1204 for ( ;; state = VFP_INITIAL )
1205 {
1206 switch ( state )
1207 {
1208 case VFP_INITIAL:
1209 got = sscanf(buf, cap_format, &live, &cap);
1210 if ( 2 != got )
1211 break;
1212 lastlive = live;
1213 lastcap = cap;
1214 state = VFP_MAYBE;
1215 return 0;
1216
1217 case VFP_MAYBE:
1218 if ( 0 != strncmp(buf, at_prefix, sizeof at_prefix - 1) )
1219 detail = nostack;
1220 else
1221 {
1222 detail = buf;
1223 state = VFP_ATE_AT;
1224 if ( NULL != strstr(buf, culprit) )
1225 return 0;
1226 }
1227 ereport(INFO, (
1228 errmsg_internal(cap_format, lastlive, lastcap),
1229 errdetail_internal("%s", detail),
1230 errhint(
1231 "To pinpoint location, set a breakpoint on this ereport "
1232 "and follow stacktrace to a functionExit(), its caller "
1233 "(a JNI method), and the immediate caller of that.")));
1234 if ( nostack == detail )
1235 continue;
1236 return 0;
1237
1238 case VFP_ATE_AT:
1239 if ( 0 == strncmp(buf, at_prefix, sizeof at_prefix - 1) )
1240 return 0; /* remain in ATE_AT state */
1241 if ( 0 != strncmp(buf, locked_prefix, sizeof locked_prefix - 1) )
1242 continue;
1243 state = VFP_ATE_LOCKED;
1244 return 0;
1245
1246 case VFP_ATE_LOCKED:
1247 if ( 0 != strncmp(buf, class_prefix, sizeof class_prefix - 1) )
1248 continue;
1249 state = VFP_ATE_AT;
1250 return 0;
1251 }
1252 break;
1253 }
1254
1255 /* Trim off trailing newline and other whitespace.
1256 */
1257 ep = bp + strlen(bp) - 1;
1258 while(ep >= bp && isspace(*ep))
1259 --ep;
1260 ++ep;
1261 *ep = 0;
1262
1263 elog(s_javaLogLevel, "%s", buf);
1264 return 0;
1265}

1267/*
1268 * Append those parts of path that has not yet been appended. The HashMap unique is
1269 * keeping track of what has been appended already. First appended part will be
1270 * prefixed with prefix.
1271 */
1272static void appendPathParts(const char* path, StringInfoData* bld, HashMap unique, const char* prefix)
1273{
1274 StringInfoData buf;
1275 if(path == 0 || strlen(path) == 0)
1276 return;
1277
1278 for (;;)
1279 {
1280 char* pathPart;
1281 char* sep;
1282 size_t len;
1283
1284 if(*path == 0)
1285 break;
1286
1287 sep = first_path_var_separator(path);
1288
1289 if(sep == path)
1290 {
1291 /* Ignore zero length components.
1292 */
1293 ++path;
1294 continue;
1295 }
1296
1297 if ( NULL == sep )
1298 len = strlen(path);
1299 else
1300 len = sep - path;
1301
1302 initStringInfo(&buf);
1303 if(*path == '$')
1304 {
1305 if( (len == 7 || first_dir_separator(path) == path + 7)
1306 && strncmp(path, "$libdir", 7) == 0)
1307 {
1308 char pathbuf[MAXPGPATH];
1309 get_pkglib_path(my_exec_path, pathbuf);
1310 len -= 7;
1311 path += 7;
1312 appendStringInfoString(&buf, pathbuf);
1313 }
1314 else
1315 ereport(ERROR, (
1316 errcode(ERRCODE_INVALID_NAME),
1317 errmsg("invalid macro name '%*s' in PL/Java module path",
1318 (int)len, path)));
1319 }
1320
1321 if(len > 0)
1322 {
1323 appendBinaryStringInfo(&buf, path, (int)len);
1324 path += len;
1325 }
1326
1327 pathPart = buf.data;
1328 if(HashMap_getByString(unique, pathPart) == 0)
1329 {
1330 if(HashMap_size(unique) == 0)
1331 appendStringInfo(bld, "%s", prefix);
1332 else
1333 appendStringInfoChar(bld, s_path_var_sep);
1334 appendStringInfo(bld, "%s", pathPart);
1335 HashMap_putByString(unique, pathPart, (void*)1);
1336 }
1337 pfree(pathPart);
1338 if(*path == 0)
1339 break;
1340 ++path; /* Skip path var separator */
1341 }
1342}

1344/*
1345 * Get the module path. Result is always freshly palloc'd.
1346 * No longer relies on an environment variable. What CLASSPATH variable might
1347 * happen to be randomly set in the environment of a PostgreSQL backend?
1348 */
1349static char* getModulePath(const char* prefix)
1350{
1351 char* path;
1352 HashMap unique = HashMap_create(13, CurrentMemoryContext);
1353 StringInfoData buf;
1354 initStringInfo(&buf);
1355 appendPathParts(modulepath, &buf, unique, prefix);

1357 path = buf.data;
1358 if(strlen(path) == 0)
1359 {
1360 pfree(path);
1361 path = 0;
1362 }
1363 return path;
1364}

1529/* Split JVM options. The string is split on whitespace unless the
1530 * whitespace is found within a string or is escaped by backslash. A
1531 * backslash escaped quote is not considered a string delimiter.
1532 */
1533static void addUserJVMOptions(JVMOptList* optList)
1534{
1535 const char* cp = vmoptions;
1536
1537 if(cp != NULL)
1538 {
1539 StringInfoData buf;
1540 char quote = 0;
1541 char c;
1542
1543 initStringInfo(&buf);
1544 for(;;)
1545 {
1546 c = *cp++;
1547 switch(c)
1548 {
1549 case 0:
1550 break;
1551
1552 case '"':
1553 case '\'':
1554 if(quote == c)
1555 quote = 0;
1556 else
1557 quote = c;
1558 appendStringInfoChar(&buf, c);
1559 continue;
1560
1561 case '\\':
1562 appendStringInfoChar(&buf, '\\');
1563 c = *cp++; /* Interpret next character verbatim */
1564 if(c == 0)
1565 break;
1566 appendStringInfoChar(&buf, c);
1567 continue;
1568
1569 default:
1570 if(quote == 0 && isspace((int)c))
1571 {
1572 while((c = *cp++) != 0)
1573 {
1574 if(!isspace((int)c))
1575 break;
1576 }
1577
1578 if(c == 0)
1579 break;
1580
1581 if(c != '-')
1582 appendStringInfoChar(&buf, ' ');
1583 else if(buf.len > 0)
1584 {
1585 /* Whitespace followed by '-' triggers new
1586 * option declaration.
1587 */
1588 JVMOptList_add(optList, buf.data, 0, true);
1589 buf.len = 0;
1590 buf.data[0] = 0;
1591 }
1592 }
1593 appendStringInfoChar(&buf, c);
1594 continue;
1595 }
1596 break;
1597 }
1598 if(buf.len > 0)
1599 JVMOptList_add(optList, buf.data, 0, true);
1600 pfree(buf.data);
1601 }

1968static Datum internalValidator(bool trusted, PG_FUNCTION_ARGS)
1969{
1970 Oid funcoid = PG_GETARG_OID(0);
1971 Invocation ctx;
1972 Oid *oidSaveLocation = NULL;
1973
1974 bool ok = CheckFunctionValidatorAccess(fcinfo->flinfo->fn_oid, funcoid);
1975 /*
1976 * CheckFunctionValidatorAccess reserves a possible future behavior where
1977 * it returns false and this validator should immediately return. Here we
1978 * abuse that convention slightly by first checking an additional constraint
1979 * on function creation in withoutEnforcing mode. That, arguably, is a check
1980 * that should never be skipped, just like the permission checks made in
1981 * CheckFunctionValidatorAccess itself.
1982 */
1983 if ( withoutEnforcement && trusted && ! superuser() )
1984 ereport(ERROR, (
1985 errmsg(
1986 "trusted PL/Java language restricted to superuser when "
1987 "\"java.security.manager\"=\"disallow\""),
1988 errdetail(
1989 "This PL/Java version enforces security policy using important "
1990 "Java features that upstream Java has disabled as of Java 24, "
1991 "as described in JEP 486. In Java 18 through 23, enforcement is "
1992 "still available, but requires "
1993 "\"-Djava.security.manager=allow\" in \"pljava.vmoptions\". "
1994 "The alternative \"-Djava.security.manager=disallow\" permits "
1995 "use on Java 24 and later, but with no enforcement and no "
1996 "distinction between trusted and untrusted. In this mode, only "
1997 "a superuser may use even a 'trusted' PL/Java language")
1998 ));
1999 if ( ! ok )
2000 PG_RETURN_VOID();
2001
2002 /*
2003 * In the call handler, which could be called heavily, funcoid gets
2004 * unconditionally stored to one of these two locations, rather than
2005 * spending extra cycles deciding whether to store it or not. A validator
2006 * will not be called as heavily, and can afford to check here whether
2007 * an Oid needs to be stored or not. The situation to avoid is where
2008 * funcoid gets stored here, as an Oid from which PL/Java's library path can
2009 * be found, but the function then gets rejected by the validator, leaving
2010 * the stored Oid invalid and useless for that purpose. Therefore, choose
2011 * here whether and where to store it, but store it only within the PG_TRY
2012 * block, and replace with InvalidOid again in the PG_CATCH.
2013 */
2014 if ( trusted )
2015 {
2016 if ( InvalidOid == pljavaTrustedOid )
2017 oidSaveLocation = &pljavaTrustedOid;
2018 }
2019 else
2020 {
2021 if ( InvalidOid == pljavaUntrustedOid )
2022 oidSaveLocation = &pljavaUntrustedOid;
2023 }
2024
2025 if ( IS_PLJAVA_INSTALLING > initstage )
2026 {
2027 if ( check_function_bodies ) /* We're gonna need a JVM */
2028 {
2029 deferInit = false;
2030 initsequencer( initstage, false);
2031 }
2032 else /* Can try to start one, but if no go, just assume function's ok */
2033 {
2034 initsequencer( initstage, true);
2035 if ( IS_PLJAVA_INSTALLING > initstage )
2036 {
2037 if ( javaGT11 )
2038 warnJEP411 = true;
2039 PG_RETURN_VOID();
2040 }
2041 }
2042 }
2043
2044 Invocation_pushInvocation(&ctx);
2045 PG_TRY();
2046 {
2047 if ( NULL != oidSaveLocation )
2048 *oidSaveLocation = funcoid;
2049
2050 Function_invoke(
2051 funcoid, trusted, false, true, check_function_bodies, NULL);
2052 Invocation_popInvocation(false);
2053 }
2054 PG_CATCH();
2055 {
2056 if ( NULL != oidSaveLocation )
2057 *oidSaveLocation = InvalidOid;
2058
2059 Invocation_popInvocation(true);
2060 PG_RE_THROW();
2061 }
2062 PG_END_TRY();
