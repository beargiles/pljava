33static Entry Iterator_peekNext(Iterator self)
34{
35 uint32 tableSize = self->source->tableSize;
36 if(tableSize != self->sourceTableSize)
37 {
38 /* Rehash during Iteration. We can't continue.
39 */
40 self->nextEntry = 0;
41 }
42 else if(self->nextEntry == 0)
43 {
44 /* Go to next bucket
45 */
46 Entry* table = self->source->table;
47 while(self->currentBucket < tableSize)
48 {
49 Entry nxt = table[self->currentBucket];
50 if(nxt != 0)
51 {
52 self->nextEntry = nxt;
53 break;
54 }
55 self->currentBucket++;
56 }
57 }
58 return self->nextEntry;
59}

61bool Iterator_hasNext(Iterator self)
62{
63 return Iterator_peekNext(self) != 0;
64}

66Entry Iterator_next(Iterator self)
67{
68 Entry nxt = Iterator_peekNext(self);
69 if(nxt != 0)
70 {
71 Entry nxtNxt = nxt->next;
72 if(nxtNxt == 0)
73 /*
74 * Leave this bucket.
75 */
76 self->currentBucket++;
77 self->nextEntry = nxtNxt;
78 }
79 return nxt;
80}
