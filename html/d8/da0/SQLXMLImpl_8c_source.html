36/*
37 * It is possible to install PL/Java in a PostgreSQL instance that was built
38 * without libxml and the native XML data type. It could even be useful for
39 * SQLXML to be usable in those circumstances, so the canReplaceType method
40 * will return true if the native type is text. (An exact match on TEXTOID is
41 * required, for now at least, because over in String.c, canReplaceType answers
42 * true for any native type that has text in/out conversions, and we do NOT want
43 * SQLXML to willy/nilly expose the internals of just any of those.
44 */
45static bool _SQLXML_canReplaceType(Type self, Type other)
46{
47 TypeClass cls = Type_getClass(other);
48 return
49 Type_getClass(self) == cls ||
50#if defined(XMLOID)
51 Type_getOid(other) == XMLOID ||
52#endif
53 Type_getOid(other) == PG_NODE_TREEOID || /* a synthetic rendering */
54 Type_getOid(other) == TEXTOID;
55}

81static Datum _SQLXML_coerceObject(Type self, jobject sqlxml)
82{
83 jobject vw = JNI_callStaticObjectMethodLocked(
84 s_SQLXML_class, s_SQLXML_adopt, sqlxml, Type_getOid(self));
85 Datum d = pljava_VarlenaWrapper_adopt(vw);
86 JNI_deleteLocalRef(vw);
87 if ( VARATT_IS_EXTERNAL_EXPANDED_RW(DatumGetPointer(d)) )
88 return TransferExpandedObject(d, CurrentMemoryContext);
89
90 MemoryContextSetParent(
91 GetMemoryChunkContext(DatumGetPointer(d)), CurrentMemoryContext);
92
93 return d;
94}

96/*
97 * A Type can be 'registered' two ways. In one case, a single instance can be
98 * created with TypeClass_allocInstance(2)? and assigned a fixed Oid, and that
99 * instance then passed to Type_registerType along with the Java name.
100 *
101 * The other way is not to allocate any Type instance up front, but instead
102 * to call Type_registerType2, passing just the type's canonical Oid, the Java
103 * name, and an 'obtainer' function, like this one.
104 *
105 * The difference appears when this TypeClass has a _canReplaceType function
106 * that allows it to serve more than one PostgreSQL type (as, indeed, SQLXML
107 * now does and can). With the first registration style, the same Type instance
108 * will be used for any of the PostgreSQL types accepted by the _canReplaceType
109 * function. With the second style, the obtainer will be called to produce a
110 * distinct Type instance (sharing the same TypeClass) for each one, recording
111 * its own PostgreSQL Oid.
112 *
113 * SQLXML has a need to run a content verifier when 'bouncing' a readable
114 * instance back to PostgreSQL, and ideally only to do so when the Oids at
115 * create and adopt time are different, so it cannot make do with the singleton
116 * type instance, and needs to use Type_registerType2 with an obtainer.
117 *
118 * The obtainer can, however, cache a single instance per supported oid, of
119 * which there are, so far, only two (one, in PG instances without XML).
120 */
121static Type _SQLXML_obtain(Oid typeId)
122{
123 static Type textInstance;
124 Oid allowedId = InvalidOid;
125 bool synthetic = false;
126 Type *cache;
127#if defined(XMLOID)
128 static Type xmlInstance;
129#endif
130 static Type pgNodeTreeInstance;
131
132 switch ( typeId )
133 {
134 case PG_NODE_TREEOID:
135 allowedId = PG_NODE_TREEOID;
136 synthetic = true;
137 cache = &pgNodeTreeInstance;
138 break;
139 default:
140 if ( TEXTOID == typeId )
141 {
142 cache = &textInstance;
143 allowedId = TEXTOID;
144 }
145#if defined(XMLOID)
146 else
147 {
148 allowedId = XMLOID;
149 cache = &xmlInstance;
150 }
151#endif
152 }
153 if ( NULL == *cache )
154 *cache = TypeClass_allocInstance(
155 synthetic ? s_SQLXMLClass_Synthetic : s_SQLXMLClass, allowedId);
156 return *cache;
157}
