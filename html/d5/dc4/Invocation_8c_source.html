32/*
33 * Two features of the calling convention for PL/Java functions will be handled
34 * here in Invocation to keep wrappers in Function simple. A PL/Java function
35 * may use static primitive slot 0 to return a primitive value, so that will
36 * always be saved in an Invocation struct and restored on both normal and
37 * exceptional return paths, when the heavier-weight full pushing of a Java
38 * ParameterFrame has not occurred. Likewise, the heavy full push is skipped if
39 * either the current or the new frame limits are (0,0), which means for such
40 * cases the frame limits themselves must be saved and restored the same way.
41 */
42static jvalue *s_primSlot0;
43static jshort *s_frameLimits;
44
45/*
46 * To keep these values somewhat encapsulated, Function.c calls this function
47 * during its initialization to share them, rather than simply making them
48 * global.
49 */
50void pljava_Invocation_shareFrame(jvalue *slot0, jshort *limits)
51{
52 if ( 0 != s_primSlot0 || 0 != s_frameLimits )
53 return;
54 s_primSlot0 = slot0;
55 s_frameLimits = limits;
56}

93void Invocation_assertConnect(void)
94{
95 int rslt;
96 if(!currentInvocation->hasConnected)
97 {
98 rslt = SPI_connect();
99 if ( SPI_OK_CONNECT != rslt )
100 elog(ERROR, "SPI_connect returned %s",
101 SPI_result_code_string(rslt));
102#if PG_VERSION_NUM >= 100000
103 if ( NULL != currentInvocation->triggerData )
104 {
105 rslt = SPI_register_trigger_data(currentInvocation->triggerData);
106 if ( SPI_OK_TD_REGISTER != rslt )
107 elog(WARNING, "SPI_register_trigger_data returned %s",
108 SPI_result_code_string(rslt));
109 }
110#endif
111 currentInvocation->hasConnected = true;
112 }
113}

137void Invocation_pushBootContext(Invocation* ctx)
138{
139 JNI_pushLocalFrame(LOCAL_FRAME_SIZE);
140 ctx->invocation = 0;
141 ctx->function = 0;
142 ctx->frameLimits = 0;
143 ctx->primSlot0.j = 0L;
144 ctx->savedLoader = 0;
145 ctx->hasConnected = false;
146 ctx->upperContext = CurrentMemoryContext;
147 ctx->errorOccurred = false;
148 ctx->inExprContextCB = false;
149 ctx->previous = 0;
150#if PG_VERSION_NUM >= 100000
151 ctx->triggerData = 0;
152#endif
153 currentInvocation = ctx;
154 ++s_callLevel;
155}

158{

163 * Nothing is done here with savedLoader. It is just set to 0 in

166 * boot context.

168}

190void Invocation_popInvocation(bool wasException)
191{
192 Invocation* ctx = currentInvocation->previous;
193 bool heavy = FRAME_LIMITS_PUSHED == currentInvocation->frameLimits;
194
195 /*
196 * If the more heavyweight parameter-frame push wasn't done, do
197 * the lighter cleanup here.
198 */
199 if ( ! heavy )
200 {
201 /*
202 * The lighter-weight cleanup.
203 */
204 *s_frameLimits = currentInvocation->frameLimits;
205 *s_primSlot0 = currentInvocation->primSlot0;
206 }
207 pljava_Function_popFrame(heavy);
208
209 /*
210 * If a Java Invocation instance was created and associated with this
211 * invocation, delete the reference (after calling its onExit method,
212 * indicating whether the return is exceptional or not).
213 */
214 if(currentInvocation->invocation != 0)
215 {
216 JNI_callVoidMethodLocked(
217 currentInvocation->invocation, s_Invocation_onExit,
218 (wasException || currentInvocation->errorOccurred)
219 ? JNI_TRUE : JNI_FALSE);
220 JNI_deleteGlobalRef(currentInvocation->invocation);
221 }
222
223 /*
224 * Do nativeRelease for any DualState instances scoped to this invocation.
225 */
226 pljava_DualState_nativeRelease(currentInvocation);
227
228 /*
229 * Check for any DualState objects that became unreachable and can be freed.
230 */
231 pljava_DualState_cleanEnqueuedInstances();
232
233 if(currentInvocation->hasConnected)
234 SPI_finish();
235
236 JNI_popLocalFrame(0);
237
238 if(ctx != 0)
239 {
240 MemoryContextSwitchTo(ctx->upperContext);
241 }
242
243 currentInvocation = ctx;
244 --s_callLevel;
245}
