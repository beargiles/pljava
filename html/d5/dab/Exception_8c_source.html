49void
50Exception_featureNotSupported(const char* requestedFeature, const char* introVersion)
51{
52 jstring jmsg;
53 jobject ex;
54 StringInfoData buf;
55 initStringInfo(&buf);
56 PG_TRY();
57 {
58 appendStringInfoString(&buf, "Feature: ");
59 appendStringInfoString(&buf, requestedFeature);
60 appendStringInfoString(&buf, " lacks support in PostgreSQL version ");
61 appendStringInfo(&buf, "%d.%d",
62 PG_VERSION_NUM / 10000,
63#if PG_VERSION_NUM >= 100000
64 (PG_VERSION_NUM) % 10000
65#else
66 (PG_VERSION_NUM / 100) % 100
67#endif
68 );
69 appendStringInfoString(&buf, ". It was introduced in version ");
70 appendStringInfoString(&buf, introVersion);
71
72 ereport(DEBUG3, (errmsg("%s", buf.data)));
73 jmsg = String_createJavaStringFromNTS(buf.data);
74
75 ex = JNI_newObject(UnsupportedOperationException_class, UnsupportedOperationException_init, jmsg);
76 JNI_deleteLocalRef(jmsg);
77 JNI_throw(ex);
78 }
79 PG_CATCH();
80 {
81 ereport(WARNING, (errcode(ERRCODE_INTERNAL_ERROR), errmsg("Exception while generating exception: %s", buf.data)));
82 }
83 PG_END_TRY();
84 pfree(buf.data);
85}

87void Exception_throw(int errCode, const char* errMessage, ...)
88{
89 char buf[1024];
90 va_list args;
91 jstring message;
92 jstring sqlState;
93 jobject ex;
94 int idx;
95
96 va_start(args, errMessage);
97 vsnprintf(buf, sizeof(buf), errMessage, args);
98 ereport(DEBUG3, (errcode(errCode), errmsg("%s", buf)));
99
100 PG_TRY();
101 {
102 message = String_createJavaStringFromNTS(buf);
103
104 /* unpack MAKE_SQLSTATE code
105 */
106 for (idx = 0; idx < 5; ++idx)
107 {
108 buf[idx] = (char)PGUNSIXBIT(errCode);
109 errCode >>= 6;
110 }
111 buf[idx] = 0;
112
113 sqlState = String_createJavaStringFromNTS(buf);
114
115 ex = JNI_newObject(SQLException_class, SQLException_init, message, sqlState);
116
117 JNI_deleteLocalRef(message);
118 JNI_deleteLocalRef(sqlState);
119 JNI_throw(ex);
120 }
121 PG_CATCH();
122 {
123 ereport(WARNING, (errcode(errCode), errmsg("Exception while generating exception: %s", buf)));
124 }
125 PG_END_TRY();
126 va_end(args);
127}

164void Exception_throw_ERROR(const char* funcName)
165{
166 jobject ex;
167 PG_TRY();
168 {
169 jobject ed = pljava_ErrorData_getCurrentError();
170
171 FlushErrorState();
172
173 ex = JNI_newObject(ServerException_class, ServerException_init, ed);
174 currentInvocation->errorOccurred = true;
175
176 elog(DEBUG2, "Exception in function %s", funcName);
177
178 JNI_deleteLocalRef(ed);
179 JNI_throw(ex);
180 }
181 PG_CATCH();
182 {
183 elog(WARNING, "Exception while generating exception");
184 }
185 PG_END_TRY();
186}
