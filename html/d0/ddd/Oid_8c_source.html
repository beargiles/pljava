37jobject Oid_create(Oid oid)
38{
39 jobject joid;
40 /*
41 * This is a natural place to have a StaticAssertStmt making sure the
42 * ubiquitous PG type 'Oid' fits in a jint. If it is ever removed from here
43 * or this code goes away, it should go someplace else. If it ever produces
44 * an error, don't assume the only things that need fixing will be in this
45 * file or nearby....
46 */
47 StaticAssertStmt(sizeof(Oid) <= sizeof(jint), "Oid wider than jint?!");
48 if(OidIsValid(oid))
49 joid = JNI_newObject(s_Oid_class, s_Oid_init, oid);
50 else
51 joid = 0;
52 return joid;
53}

62Oid Oid_forSqlType(int sqlType)
63{
64 Oid typeId;
65
66 switch(sqlType)
67 {
68 case java_sql_Types_BIT:
69 typeId = BITOID;
70 break;
71 case java_sql_Types_TINYINT:
72 typeId = CHAROID;
73 break;
74 case java_sql_Types_SMALLINT:
75 typeId = INT2OID;
76 break;
77 case java_sql_Types_INTEGER:
78 typeId = INT4OID;
79 break;
80 case java_sql_Types_BIGINT:
81 typeId = INT8OID;
82 break;
83 case java_sql_Types_FLOAT:
84 case java_sql_Types_REAL:
85 typeId = FLOAT4OID;
86 break;
87 case java_sql_Types_DOUBLE:
88 typeId = FLOAT8OID;
89 break;
90 case java_sql_Types_NUMERIC:
91 case java_sql_Types_DECIMAL:
92 typeId = NUMERICOID;
93 break;
94 case java_sql_Types_DATE:
95 typeId = DATEOID;
96 break;
97 case java_sql_Types_TIME:
98 typeId = TIMEOID;
99 break;
100 case java_sql_Types_TIMESTAMP:
101 typeId = TIMESTAMPOID;
102 break;
103 case java_sql_Types_BOOLEAN:
104 typeId = BOOLOID;
105 break;
106 case java_sql_Types_BINARY:
107 case java_sql_Types_VARBINARY:
108 case java_sql_Types_LONGVARBINARY:
109 case java_sql_Types_BLOB:
110 typeId = BYTEAOID;
111 break;
112 case java_sql_Types_CHAR:
113 case java_sql_Types_VARCHAR:
114 case java_sql_Types_LONGVARCHAR:
115 case java_sql_Types_CLOB:
116 case java_sql_Types_DATALINK:
117 typeId = TEXTOID;
118 break;
119 case java_sql_Types_NULL:
120 case java_sql_Types_OTHER:
121 case java_sql_Types_JAVA_OBJECT:
122 case java_sql_Types_DISTINCT:
123 case java_sql_Types_STRUCT:
124 case java_sql_Types_ARRAY:
125 case java_sql_Types_REF:
126 typeId = InvalidOid; /* Not yet mapped */
127 break;
128
129 /* JDBC 4.0 - present in Java 6 and later, no need to conditionalize */
130 case java_sql_Types_SQLXML:
131#ifdef XMLOID /* but PG can have been built without libxml */
132 typeId = XMLOID;
133#else
134 typeId = InvalidOid;
135#endif
136 break;
137 case java_sql_Types_ROWID:
138 case java_sql_Types_NCHAR:
139 case java_sql_Types_NVARCHAR:
140 case java_sql_Types_LONGNVARCHAR:
141 case java_sql_Types_NCLOB:
142 typeId = InvalidOid; /* Not yet mapped */
143 break;
144
145 case java_sql_Types_TIME_WITH_TIMEZONE:
146 typeId = TIMETZOID;
147 break;
148 case java_sql_Types_TIMESTAMP_WITH_TIMEZONE:
149 typeId = TIMESTAMPTZOID;
150 break;
151 case java_sql_Types_REF_CURSOR:
152 default:
153 typeId = InvalidOid; /* Not yet mapped */
154 break;
155 }
156 return typeId;
157}
