40jobject pljava_Portal_create(Portal portal, jobject jplan)
41{
42 jobject jportal;
43 Ptr2Long p2l;
44 Ptr2Long p2lro;
45 if(portal == 0)
46 return NULL;
47
48 p2l.longVal = 0L; /* ensure that the rest is zeroed out */
49 p2l.ptrVal = portal;
50
51 p2lro.longVal = 0L;
52 p2lro.ptrVal = portal->resowner;
53
54 jportal = JNI_newObjectLocked(s_Portal_class, s_Portal_init,
55 pljava_DualState_key(), p2lro.longVal, p2l.longVal, jplan);
56
57 return jportal;
58}

140 jlong result = 0;
141 if(_this != 0)
142 {
143 BEGIN_NATIVE
144 Ptr2Long p2l;
145 STACK_BASE_VARS
146 STACK_BASE_PUSH(env)
147
148 /*
149 * One call to cleanEnqueued... is made in Invocation_popInvocation,
150 * when any PL/Java function returns to PostgreSQL. But what of a
151 * PL/Java function that loops through a lot of data before returning?
152 * It could be important to call cleanEnqueued... from some other
153 * strategically-chosen places, and this seems a good one. We get here
154 * every fetchSize (default 1000? See SPIStatement) rows retrieved.
155 */
156 pljava_DualState_cleanEnqueuedInstances();
157
158 p2l.longVal = _this;
159 PG_TRY();
160 {
161 Invocation_assertConnect();
162 SPI_cursor_fetch((Portal)p2l.ptrVal, forward == JNI_TRUE,
163 (long)count);
164 result = (jlong)SPI_processed;
165 }
166 PG_CATCH();
167 {
168 Exception_throw_ERROR("SPI_cursor_fetch");
169 }
170 PG_END_TRY();
171 STACK_BASE_POP()
172 END_NATIVE
173 }
174 return result;
175}
