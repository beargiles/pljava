17void arraySetNull(bits8* bitmap, int offset, bool flag)
18{
19 if(bitmap != 0)
20 {
21 int bitmask = 1 << (offset % 8);
22 bitmap += offset / 8;
23 *bitmap = (bits8)(flag? *bitmap & ~bitmask : *bitmap | bitmask);
24 }
25}

27bool arrayIsNull(const bits8* bitmap, int offset)
28{
29 return bitmap == 0 ? false : !(bitmap[offset / 8] & (1 << (offset % 8)));
30}

32ArrayType* createArrayType(jsize nElems, size_t elemSize, Oid elemType, bool withNulls)
33{
34 ArrayType* v;
35 Size nBytes = elemSize * nElems;
36 MemoryContext currCtx = Invocation_switchToUpperContext();
37
38 Size dataoffset;
39 if(withNulls)
40 {
41 dataoffset = ARR_OVERHEAD_WITHNULLS(1, nElems);
42 nBytes += dataoffset;
43 }
44 else
45 {
46 dataoffset = 0; /* marker for no null bitmap */
47 nBytes += ARR_OVERHEAD_NONULLS(1);
48 }
49 v = (ArrayType*)palloc0(nBytes);
50 AssertVariableIsOfType(v->dataoffset, int32);
51 v->dataoffset = (int32)dataoffset;
52 MemoryContextSwitchTo(currCtx);
53
54 SET_VARSIZE(v, nBytes);
55 ARR_NDIM(v) = 1;
56 ARR_ELEMTYPE(v) = elemType;
57 *((int*)ARR_DIMS(v)) = nElems;
58 *((int*)ARR_LBOUND(v)) = 1;
59 return v;
60}

62static jvalue _Array_coerceDatum(Type self, Datum arg)
63{
64 jvalue result;
65 jsize idx;
66 Type elemType = Type_getElementType(self);
67 int16 elemLength = Type_getLength(elemType);
68 char elemAlign = Type_getAlign(elemType);
69 bool elemByValue = Type_isByValue(elemType);
70 ArrayType* v = DatumGetArrayTypeP(arg);
71 jsize nElems = (jsize)ArrayGetNItems(ARR_NDIM(v), ARR_DIMS(v));
72 jobjectArray objArray = JNI_newObjectArray(nElems, Type_getJavaClass(elemType), 0);
73 const char* values = ARR_DATA_PTR(v);
74 bits8* nullBitMap = ARR_NULLBITMAP(v);
75
76 for(idx = 0; idx < nElems; ++idx)
77 {
78 if(arrayIsNull(nullBitMap, idx))
79 JNI_setObjectArrayElement(objArray, idx, 0);
80 else
81 {
82 Datum value = fetch_att(values, elemByValue, elemLength);
83 jvalue obj = Type_coerceDatum(elemType, value);
84 JNI_setObjectArrayElement(objArray, idx, obj.l);
85 JNI_deleteLocalRef(obj.l);
86
87 values = att_addlength_datum(values, elemLength, PointerGetDatum(values));
88 values = (char*)att_align_nominal(values, elemAlign);
89 }
90 }
91 result.l = (jobject)objArray;
92 return result;
93}

95static Datum _Array_coerceObject(Type self, jobject objArray)
96{
97 ArrayType* v;
98 jsize idx;
99 int lowerBound = 1;
100 Type elemType = Type_getElementType(self);
101 int nElems = (int)JNI_getArrayLength((jarray)objArray);
102 Datum* values = (Datum*)palloc(nElems * sizeof(Datum) + nElems * sizeof(bool));
103 bool* nulls = (bool*)(values + nElems);
104
105 for(idx = 0; idx < nElems; ++idx)
106 {
107 jobject obj = JNI_getObjectArrayElement(objArray, idx);
108 if(obj == 0)
109 {
110 nulls[idx] = true;
111 values[idx] = 0;
112 }
113 else
114 {
115 nulls[idx] = false;
116 values[idx] = Type_coerceObject(elemType, obj);
117 JNI_deleteLocalRef(obj);
118 }
119 }
120
121 v = construct_md_array(
122 values,
123 nulls,
124 1,
125 &nElems,
126 &lowerBound,
127 Type_getOid(elemType),
128 Type_getLength(elemType),
129 Type_isByValue(elemType),
130 Type_getAlign(elemType));
131
132 pfree(values);
133 PG_RETURN_ARRAYTYPE_P(v);
134}

136/*
137 * For an array, canReplaceType can be computed a bit more generously.
138 * The primitive types are coded so that a boxed scalar can replace its
139 * corresponding primitive but not vice versa. For primitive arrays, we can also
140 * accept the other direction, that is, when getObjectType(self) == other. That
141 * will work because every primitive Type foo does contain _fooArray_coerceDatum
142 * and _fooArray_coerceObject and can handle both directions.
143 */
144static bool _Array_canReplaceType(Type self, Type other)
145{
146 Type oe = Type_getElementType(other);
147 if ( oe == 0 )
148 return false;
149 return Type_canReplaceType(Type_getElementType(self), oe)
150 || Type_getObjectType(self) == other;
151}

158Type Array_fromOid2(Oid typeId, Type elementType, DatumCoercer coerceDatum, ObjectCoercer coerceObject)
159{
160 Type self;
161 TypeClass arrayClass;
162 const char* elemClassName = PgObjectClass_getName(PgObject_getClass((PgObject)elementType));
163 const char* elemJNISignature = Type_getJNISignature(elementType);
164 const char* elemJavaTypeName = Type_getJavaTypeName(elementType);
165
166 MemoryContext currCtx = MemoryContextSwitchTo(TopMemoryContext);
167
168 char* tmp = palloc(strlen(elemClassName) + 3);
169 sprintf(tmp, "%s[]", elemClassName);
170 arrayClass = TypeClass_alloc(tmp);
171
172 tmp = palloc(strlen(elemJNISignature) + 2);
173 sprintf(tmp, "[%s", elemJNISignature);
174 arrayClass->JNISignature = tmp;
175
176 tmp = palloc(strlen(elemJavaTypeName) + 3);
177 sprintf(tmp, "%s[]", elemJavaTypeName);
178 arrayClass->javaTypeName = tmp;
179 arrayClass->coerceDatum = coerceDatum;
180 arrayClass->coerceObject = coerceObject;
181 arrayClass->canReplaceType = _Array_canReplaceType;
182 self = TypeClass_allocInstance(arrayClass, typeId);
183 MemoryContextSwitchTo(currCtx);
184
185 self->elementType = elementType;
186 Type_registerType(arrayClass->javaTypeName, self);
187
188 if(Type_isPrimitive(elementType))
189 self->objectType = Array_fromOid(typeId, Type_getObjectType(elementType));
190 return self;
191}
