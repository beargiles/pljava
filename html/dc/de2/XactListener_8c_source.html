22static void xactCB(XactEvent event, void* arg)
23{
24 /*
25 * Upstream has, regrettably, not merely added events over the years, but
26 * changed their order, so a mapping is needed. Use a switch with the known
27 * cases enumerated, to improve the chance that a clever compiler will warn
28 * if yet more have been added, and initialize 'mapped' to a value that the
29 * Java code won't mistake for a real one.
30 */
31#define CASE(c) \
32case XACT_EVENT_##c: \
33 mapped = org_postgresql_pljava_internal_XactListener_##c; \
34 break
35
36 jint mapped = -1;
37 switch(event)
38 {
39 CASE( COMMIT );
40 CASE( ABORT );
41 CASE( PREPARE );
42 CASE( PRE_COMMIT );
43 CASE( PRE_PREPARE );
44 CASE( PARALLEL_COMMIT );
45 CASE( PARALLEL_ABORT );
46 CASE( PARALLEL_PRE_COMMIT );
47 }
48
49 JNI_callStaticVoidMethod(s_XactListener_class,
50 s_XactListener_invokeListeners, mapped);
51}
