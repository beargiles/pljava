122JNIEXPORT jint JNICALL
123Java_org_postgresql_pljava_internal_SPI__1exec(JNIEnv* env, jclass cls, jstring cmd, jint count)
124{
125 jint result = 0;
126
127 BEGIN_NATIVE
128 char* command = String_createNTS(cmd);
129 if(command != 0)
130 {
131 STACK_BASE_VARS
132 STACK_BASE_PUSH(env)
133 PG_TRY();
134 {
135 Invocation_assertConnect();
136 result = (jint)SPI_exec(command, (int)count);
137 if(result < 0)
138 Exception_throwSPI("exec", result);
139
140 }
141 PG_CATCH();
142 {
143 Exception_throw_ERROR("SPI_exec");
144 }
145 PG_END_TRY();
146 pfree(command);
147 STACK_BASE_POP()
148 }
149 END_NATIVE
150 return result;
151}
