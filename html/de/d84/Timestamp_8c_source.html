26/*
27 * Types timestamp and timestamptz. This compilation unit supplies code for both
28 * PostgreSQL types. The legacy JDBC mapping for both is to java.sql.Timestamp,
29 * which holds an implicit timezone offset and therefore can't be an equally
30 * good fit for both.
31 *
32 * Java 8 and JDBC 4.2 introduce java.time.LocalDateTime and
33 * java.time.OffsetDateTime, which more directly fit PG's timestamp and
34 * timestamptz, respectively. For compatibility reasons, the legacy behavior of
35 * getObject (with no Class parameter) is unchanged, and still returns the data
36 * weirdly shoehorned into java.sql.Timestamp. But Java 8 application code can
37 * and should use the form of getObject with a Class parameter to request
38 * java.time.LocalDateTime or java.time.OffsetDateTime, as appropriate.
39 *
40 * Note that it is somewhat misleading for PostgreSQL to call one of these types
41 * TIMESTAMP WITH TIME ZONE. The stored form does not, in fact, include a time
42 * zone (and this is in contrast to TIME WITH TIME ZONE, which does). Instead,
43 * what PostgreSQL means by TIMESTAMP WITH TIMEZONE is that a zone can be given
44 * (or inferred from the session) when a value is input, and used to adjust the
45 * value to UTC, and, likewise, the stored UTC value can be output converted to
46 * a given (or implicit) zone offset. Meanwhile, a TIMESTAMP WITHOUT TIME ZONE
47 * is just stored as the number of seconds from epoch that would make a clock
48 * on UTC show the same date and time as the value input.
49 *
50 * When producing a java.time.LocalDateTime from a timestamp and vice versa,
51 * the conversion is just what you would think. When producing an OffsetDateTime
52 * from a timestamptz, the OffsetDateTime will always have offset zero from UTC.
53 * That's what the stored PostgreSQL data represents; to produce anything else
54 * would be lying. When receiving an OffsetDateTime into PostgreSQL, of course
55 * any zone offset it contains will be used to adjust the value to UTC for
56 * storage.
57 *
58 * The legacy behavior when mapping timestamp and timestamptz to
59 * java.sql.Timestamp is that a timestamptz is converted in both directions
60 * without alteration, and a (local!) timestamp is *adjusted as if to UTC from
61 * the current session's implicit timezone* (and vice versa when receiving
62 * a value). Weird or not, that's how PL/Java has always done it.
63 */

107static jvalue _LocalDateTime_coerceDatum(Type self, Datum arg)
108{
109 jint onlyMicros;
110 jlong secs;
111 jvalue result;
112#if PG_VERSION_NUM < 100000
113 if ( !integerDateTimes )
114 {
115 double fracSecs = DatumGetFloat8(arg);
116 secs = (jlong)floor(fracSecs);
117 fracSecs -= (double)secs;
118 onlyMicros = ((jint)floor(2e6 * fracSecs) + 1) / 2;
119 }
120 else
121#endif
122 {
123 int64 micros = DatumGetInt64(arg);
124 /* Expect number of microseconds since 01 Jan 2000. Tease out a
125 * non-negative sub-second microseconds value (whether this C compiler's
126 * signed % has trunc or floor behavior). Factor a 2 out right away to
127 * avoid wraparound when flooring near the most negative values.
128 */
129 int lowBit = (int)(micros & 1);
130 micros = (micros ^ lowBit) / 2;
131 onlyMicros = (jint)(((micros % 500000) + 500000) % 500000);
132 secs = (micros - onlyMicros) / 500000;
133 onlyMicros = (onlyMicros << 1) | lowBit;
134 }
135 result.l = JNI_callStaticObjectMethod(s_LocalDateTime_class,
136 s_LocalDateTime_ofEpochSecond,
137 EPOCH_DIFF + secs, 1000 * onlyMicros, s_ZoneOffset_UTC);
138 return result;
139}

205static Datum _OffsetDateTime_coerceObject(Type self, jobject timestamp)
206{
207 jlong epochSec = JNI_callLongMethod(
208 timestamp, s_OffsetDateTime_toEpochSecond) - EPOCH_DIFF;
209 jint nanos = JNI_callIntMethod(timestamp, s_OffsetDateTime_getNano);
210 Datum result;
211
212#if PG_VERSION_NUM < 100000
213 if ( !integerDateTimes )
214 {
215 double secs = (double)epochSec + ((double)nanos)/1e9;
216 result = Float8GetDatum(secs);
217 }
218 else
219#endif
220 {
221 result = Int64GetDatum(1000000L * epochSec + nanos / 1000);
222 }
223
224 return result;
225}

251static jvalue Timestamp_coerceDatumTZ_id(Type self, Datum arg, bool tzAdjust)
252{
253 jvalue result;
254 jint uSecs;
255 jlong mSecs;
256 int64 ts = DatumGetInt64(arg);
257
258 /* Expect number of microseconds since 01 Jan 2000. Tease out a non-negative
259 * sub-second microseconds value (whether this C compiler's signed %
260 * has trunc or floor behavior). Factor a 2 out right away to
261 * avoid wraparound when flooring near the most negative values.
262 */
263 int lowBit = (int)(ts & 1);
264 ts = (ts ^ lowBit) / 2;
265 uSecs = (jint)(((ts % 500000) + 500000) % 500000);
266 mSecs = (ts - uSecs) / 500; /* Convert to millisecs */
267 uSecs = (uSecs << 1) | lowBit;
268
269 if(tzAdjust)
270 {
271 int tz = Timestamp_getTimeZone_id(ts); /* function expects halved ts */
272 mSecs += tz * 1000; /* Adjust from local time to UTC */
273 }
274
275 /* Adjust for diff between Postgres and Java (Unix) */
276 mSecs += ((jlong)EPOCH_DIFF) * 1000L;
277
278 result.l = JNI_newObject(s_Timestamp_class, s_Timestamp_init, mSecs);
279 if(uSecs != 0)
280 JNI_callVoidMethod(result.l, s_Timestamp_setNanos, uSecs * 1000);
281 return result;
282}

285static jvalue Timestamp_coerceDatumTZ_dd(Type self, Datum arg, bool tzAdjust)
286{
287 jlong secs;
288 jint uSecs;
289 jvalue result;
290 double ts = DatumGetFloat8(arg);
291 int tz = Timestamp_getTimeZone_dd(ts);
292
293 /* Expect <seconds since Jan 01 2000>.<fractions of seconds>
294 */
295 if(tzAdjust)
296 ts += tz; /* Adjust from local time to UTC */
297 ts += EPOCH_DIFF; /* Adjust for diff between Postgres and Java (Unix) */
298 secs = (jlong) floor(ts); /* Take just the secs */
299 uSecs = (((jint) ((ts - (double)secs) * 2e6)) + 1) / 2; /* Preserve usecs */
300 result.l = JNI_newObject(s_Timestamp_class, s_Timestamp_init, secs * 1000);
301 if(uSecs != 0)
302 JNI_callVoidMethod(result.l, s_Timestamp_setNanos, uSecs * 1000);
303 return result;
304}
305#endif

316static Datum Timestamp_coerceObjectTZ_id(Type self, jobject jts, bool tzAdjust)
317{
318 int64 ts;
319 int lowBit;
320 jlong mSecs = JNI_callLongMethod(jts, s_Timestamp_getTime);
321 jint nSecs = JNI_callIntMethod(jts, s_Timestamp_getNanos);
322 /*
323 * getNanos() should have supplied non-negative nSecs, whether mSecs is
324 * positive or negative. So mSecs needs to be floor()ed to a multiple of
325 * 1000 ms, whether this C compiler does signed integer division with floor
326 * or trunc.
327 */
328 mSecs -= ((mSecs % 1000) + 1000) % 1000;
329 mSecs -= ((jlong)EPOCH_DIFF) * 1000L;
330 ts = mSecs * 500L; /* millisecs to microsecs, save a factor of 2 for now */
331 if(tzAdjust) /* Adjust from UTC to local time; function expects halved ts */
332 ts -= ((jlong)Timestamp_getTimeZone_id(ts)) * 500000L;
333 nSecs /= 1000; /* ok, now they are really microsecs */
334 lowBit = nSecs & 1;
335 nSecs >>= 1; /* nSecs >= 0 so >> has a defined C result */
336 ts = 2 * (ts + nSecs) | lowBit;
337 return Int64GetDatum(ts);
338}

340#if PG_VERSION_NUM < 100000
341static Datum Timestamp_coerceObjectTZ_dd(Type self, jobject jts, bool tzAdjust)
342{
343 double ts;
344 jlong mSecs = JNI_callLongMethod(jts, s_Timestamp_getTime);
345 jint nSecs = JNI_callIntMethod(jts, s_Timestamp_getNanos);
346 /*
347 * getNanos() should have supplied non-negative nSecs, whether mSecs is
348 * positive or negative. So mSecs needs to be floor()ed to a multiple of
349 * 1000 ms, whether this C compiler does signed integer division with floor
350 * or trunc.
351 */
352 mSecs -= ((mSecs % 1000) + 1000) % 1000;
353 ts = ((double)mSecs) / 1000.0; /* Convert to seconds */
354 ts -= EPOCH_DIFF;
355 if(nSecs != 0)
356 ts += ((double)nSecs) / 1000000000.0; /* Convert to seconds */
357 if(tzAdjust)
358 ts -= Timestamp_getTimeZone_dd(ts); /* Adjust from UTC to local time */
359 return Float8GetDatum(ts);
360}
361#endif

402/*
403 * The argument to this function is in seconds from the PostgreSQL epoch, and
404 * the return is a time zone offset in seconds west of Greenwich.
405 */
406static int32 Timestamp_getTimeZone(pg_time_t time)
407{
408 struct pg_tm* tx = pg_localtime(&time, session_timezone);
409 if ( NULL == tx )
410 ereport(ERROR, (
411 errcode(ERRCODE_DATA_EXCEPTION),
412 errmsg("could not resolve timestamp: %m")
413 ));
414 return -(int32)tx->tm_gmtoff;
415}

417/*
418 * This is only used here and in Date.c. The caller must know that the argument
419 * is not a PostgreSQL int64 Timestamp, but, rather, one of those divided by 2.
420 */
421int32 Timestamp_getTimeZone_id(int64 dt)
422{
423 return Timestamp_getTimeZone(
424 dt / INT64CONST(500000) +
425 (POSTGRES_EPOCH_JDATE - UNIX_EPOCH_JDATE) * 86400
426 );
427}
