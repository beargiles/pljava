26/*
27 * Types time and timetz. This compilation unit supplies code for both
28 * PostgreSQL types. The legacy JDBC mapping for both is to java.sql.Time, which
29 * holds an implicit timezone offset and therefore can't be an equally good fit
30 * for both. Also, it loses precision: PostgreSQL maintains microseconds, but
31 * java.sql.Time only holds milliseconds.
32 *
33 * Java 8 and JDBC 4.2 introduce java.time.LocalTime and java.time.OffsetTime,
34 * which directly fit PG's time and timetz, respectively. For compatibility
35 * reasons, the legacy behavior of getObject (with no Class parameter) is
36 * unchanged, and still returns the data weirdly shoehorned into java.sql.Time.
37 * But Java 8 application code can and should use the form of getObject with a
38 * Class parameter to request java.time.LocalTime or java.time.OffsetTime, as
39 * appropriate.
40 *
41 * The legacy shoehorning adjusts the PostgreSQL-maintained time by its
42 * associated offset (in the timetz case), or by the current value of the server
43 * timezone offset (in the time case). Which convention is weirder?
44 */

78static jvalue _LocalTime_coerceDatum(Type self, Datum arg)
79{
80 jlong nanos =
81#if PG_VERSION_NUM < 100000
82 (!integerDateTimes) ? (jlong)floor(1e9 * DatumGetFloat8(arg)) :
83#endif
84 1000 * DatumGetInt64(arg);
85 jvalue result;
86 if ( 1000L * USECS_PER_DAY == nanos )
87 -- nanos;
88 result.l = JNI_callStaticObjectMethod(
89 s_LocalTime_class, s_LocalTime_ofNanoOfDay, nanos);
90 return result;
91}

93static Datum _LocalTime_coerceObject(Type self, jobject time)
94{
95 jlong nanos = JNI_callLongMethod(time, s_LocalTime_toNanoOfDay);
96 return
97#if PG_VERSION_NUM < 100000
98 (!integerDateTimes) ? Float8GetDatum(((double)nanos) / 1e9) :
99#endif
100 Int64GetDatum((nanos + 1) / 1000);
101}

130static jvalue _OffsetTime_coerceDatum(Type self, Datum arg)
131{
132 jvalue localTime;
133 jobject zoneOffset;
134 int32 offsetSecs;
135 jvalue result;
136
137#if PG_VERSION_NUM < 100000
138 if ( !integerDateTimes )
139 {
140 TimeTzADT_dd* tza = (TimeTzADT_dd*)DatumGetPointer(arg);
141 localTime =
142 Type_coerceDatum(s_LocalTimeInstance, Float8GetDatum(tza->time));
143 offsetSecs = tza->zone;
144 }
145 else
146#endif
147 {
148 TimeTzADT_id* tza = (TimeTzADT_id*)DatumGetPointer(arg);
149 localTime =
150 Type_coerceDatum(s_LocalTimeInstance, Int64GetDatum(tza->time));
151 offsetSecs = tza->zone;
152 }
153
154 zoneOffset = JNI_callStaticObjectMethod(s_ZoneOffset_class,
155 s_ZoneOffset_ofTotalSeconds, - offsetSecs); /* PG/Java signs differ */
156
157 result.l = JNI_callStaticObjectMethod(
158 s_OffsetTime_class, s_OffsetTime_of, localTime.l, zoneOffset);
159
160 JNI_deleteLocalRef(localTime.l);
161 JNI_deleteLocalRef(zoneOffset);
162
163 return result;
164}

166static Datum _OffsetTime_coerceObject(Type self, jobject time)
167{
168 jobject localTime = JNI_callObjectMethod(time, s_OffsetTime_toLocalTime);
169 jobject zoneOffset = JNI_callObjectMethod(time, s_OffsetTime_getOffset);
170 jint offsetSecs =
171 - /* PG/Java signs differ */
172 JNI_callIntMethod(zoneOffset, s_ZoneOffset_getTotalSeconds);
173 Datum result;
174
175#if PG_VERSION_NUM < 100000
176 if ( !integerDateTimes )
177 {
178 TimeTzADT_dd* tza = (TimeTzADT_dd*)palloc(sizeof(TimeTzADT_dd));
179 tza->zone = offsetSecs;
180 tza->time =
181 DatumGetFloat8(Type_coerceObject(s_LocalTimeInstance, localTime));
182 result = PointerGetDatum(tza);
183 }
184 else
185#endif
186 {
187 TimeTzADT_id* tza = (TimeTzADT_id*)palloc(sizeof(TimeTzADT_id));
188 tza->zone = offsetSecs;
189 tza->time =
190 DatumGetInt64(Type_coerceObject(s_LocalTimeInstance, localTime));
191 result = PointerGetDatum(tza);
192 }
193
194 JNI_deleteLocalRef(localTime);
195 JNI_deleteLocalRef(zoneOffset);
196 return result;
197}

228static jlong msecsAtMidnight(void)
229{
230 pg_time_t now = (pg_time_t)time(NULL) / 86400;
231 return INT64CONST(1000) * (jlong)(now * 86400);
232}

235static jvalue Time_coerceDatumTZ_dd(Type self, double t, bool tzAdjust)
236{
237 jlong mSecs;
238 jvalue result;
239 if(tzAdjust)
240 t += Timestamp_getCurrentTimeZone();/* Adjust from local time to UTC */
241 t *= 1000.0; /* Convert to millisecs */
242 mSecs = (jlong)floor(t);
243 result.l = JNI_newObject(s_Time_class, s_Time_init, mSecs + msecsAtMidnight());
244 return result;
245}

248static jvalue Time_coerceDatumTZ_id(Type self, int64 t, bool tzAdjust)
249{
250 jvalue result;
251 jlong mSecs = t / 1000; /* Convert to millisecs */
252 if(tzAdjust)
253 mSecs += Timestamp_getCurrentTimeZone() * 1000;/* Adjust from local time to UTC */
254 result.l = JNI_newObject(s_Time_class, s_Time_init, mSecs + msecsAtMidnight());
255 return result;
256}

258static jlong Time_getMillisecsToday(Type self, jobject jt, bool tzAdjust)
259{
260 jlong mSecs = JNI_callLongMethod(jt, s_Time_getTime);
261 if(tzAdjust)
262 mSecs -= ((jlong)Timestamp_getCurrentTimeZone()) * 1000L; /* Adjust from UTC to local time */
263 mSecs %= 86400000; /* Strip everything above 24 hours */
264 return mSecs;
265}

301/*
302 * Time with time zone. Postgres will pass local time and an associated
303 * time zone. In the future, we might create a special java object for
304 * this. For now, we just convert to UTC and pass a Time object.
305 */
306static jvalue _Timetz_coerceDatum(Type self, Datum arg)
307{
308 jvalue val;
309#if PG_VERSION_NUM < 100000
310 if(!integerDateTimes)
311 {
312 TimeTzADT_dd* tza = (TimeTzADT_dd*)DatumGetPointer(arg);
313 double t = tza->time + tza->zone; /* Convert to UTC */
314 val = Time_coerceDatumTZ_dd(self, t, false);
315 }
316 else
317#endif
318 {
319 TimeTzADT_id* tza = (TimeTzADT_id*)DatumGetPointer(arg);
320 int64 t = tza->time + (int64)tza->zone * 1000000; /* Convert to UTC */
321 val = Time_coerceDatumTZ_id(self, t, false);
322 }
323 return val;
324}

326static Datum _Timetz_coerceObject(Type self, jobject time)
327{
328 Datum datum;
329#if PG_VERSION_NUM < 100000
330 if(!integerDateTimes)
331 {
332 TimeTzADT_dd* tza = (TimeTzADT_dd*)palloc(sizeof(TimeTzADT_dd));
333 tza->time = Time_coerceObjectTZ_dd(self, time, false);
334 tza->zone = Timestamp_getCurrentTimeZone();
335 tza->time -= tza->zone; /* Convert UTC to local time */
336 datum = PointerGetDatum(tza);
337 }
338 else
339#endif
340 {
341 TimeTzADT_id* tza = (TimeTzADT_id*)palloc(sizeof(TimeTzADT_id));
342 tza->time = Time_coerceObjectTZ_id(self, time, false);
343 tza->zone = Timestamp_getCurrentTimeZone();
344 tza->time -= (int64)tza->zone * 1000000; /* Convert UTC to local time */
345 datum = PointerGetDatum(tza);
346 }
347 return datum;
348}
