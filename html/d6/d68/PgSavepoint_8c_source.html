25/*
26 * Workaround for issue #260, PostgreSQL API breakage by EnterpriseDB. They
27 * added a ReleaseCurrentSubTransactionEx function with an added argument, and
28 * made ReleaseCurrentSubTransaction call it, passing false. But instead of
29 * leaving ReleaseCurrentSubTransaction an actual function that does so, which
30 * would not have been an API break, they made it a macro instead, with the
31 * result that its address cannot be taken. The reporter of the issue had an
32 * inquiry open with EDB for four months trying to get specifics on what
33 * versions have that issue, with no useful response. So this workaround is just
34 * conditioned on finding ReleaseCurrentSubTransaction defined as a macro.
35 */
36#ifdef ReleaseCurrentSubTransaction
37static void addressableRelease(void);
38static void addressableRelease()
39{
40 ReleaseCurrentSubTransaction();
41}
42#undef ReleaseCurrentSubTransaction
43#define ReleaseCurrentSubTransaction addressableRelease
44#endif

100static void unwind(void (*f)(void), jint xid, jint nestingLevel)
101{
102 while ( nestingLevel < GetCurrentTransactionNestLevel() )
103 f();
104
105 if ( nestingLevel == GetCurrentTransactionNestLevel() )
106 {
107 assertXid((SubTransactionId)xid);
108 f();
109 }
110}
111
112static void assertXid(SubTransactionId xid)
113{
114 if(xid != GetCurrentSubTransactionId())
115 {
116 /* Oops. Rollback to top level transaction.
117 */
118 ereport(ERROR, (
119 errcode(ERRCODE_INVALID_TRANSACTION_TERMINATION),
120 errmsg("Subtransaction mismatch at txlevel %d",
121 GetCurrentTransactionNestLevel())));
122 }
123}

133JNIEXPORT jint JNICALL
134Java_org_postgresql_pljava_internal_PgSavepoint__1set(JNIEnv* env, jobject this, jstring jname)
135{
136 jint xid = 0;
137 BEGIN_NATIVE
138 PG_TRY();
139 {
140 char* name = String_createNTS(jname);
141 Invocation_assertConnect();
142 JNI_setIntField(this, s_nestLevel, 1+GetCurrentTransactionNestLevel());
143 BeginInternalSubTransaction(name);
144 xid = GetCurrentSubTransactionId();
145 if ( NULL != name )
146 pfree(name);
147 }
148 PG_CATCH();
149 {
150 Exception_throw_ERROR("setSavepoint");
151 }
152 PG_END_TRY();
153 END_NATIVE
154 return xid;
155}
