39static jvalue _booleanArray_coerceDatum(Type self, Datum arg)
40{
41 jvalue result;
42 ArrayType* v = DatumGetArrayTypeP(arg);
43 jsize nElems = (jsize)ArrayGetNItems(ARR_NDIM(v), ARR_DIMS(v));
44 jbooleanArray booleanArray = JNI_newBooleanArray(nElems);
45
46 if(ARR_HASNULL(v))
47 {
48 jsize idx;
49 jboolean isCopy = JNI_FALSE;
50 bits8* nullBitMap = ARR_NULLBITMAP(v);
51 jboolean* values = (jboolean*)ARR_DATA_PTR(v);
52 jboolean* elems = JNI_getBooleanArrayElements(booleanArray, &isCopy);
53 for(idx = 0; idx < nElems; ++idx)
54 {
55 if(arrayIsNull(nullBitMap, idx))
56 elems[idx] = 0;
57 else
58 elems[idx] = *values++;
59 }
60 JNI_releaseBooleanArrayElements(booleanArray, elems, JNI_COMMIT);
61 }
62 else
63 JNI_setBooleanArrayRegion(booleanArray, 0, nElems, (jboolean*)ARR_DATA_PTR(v));
64 result.l = (jobject)booleanArray;
65 return result;
66}
