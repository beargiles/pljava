26jobject pljava_ErrorData_getCurrentError(void)
27{
28 Ptr2Long p2l;
29 jobject jed;
30
31 MemoryContext curr = MemoryContextSwitchTo(JavaMemoryContext);
32 ErrorData* errorData = CopyErrorData();
33 MemoryContextSwitchTo(curr);
34
35 p2l.longVal = 0L; /* ensure that the rest is zeroed out */
36 p2l.ptrVal = errorData;
37 /*
38 * Passing (jlong)0 as the ResourceOwner means this will never be matched by
39 * a nativeRelease call; that's appropriate (for now) as the ErrorData copy
40 * is being made into JavaMemoryContext, which never gets reset, so only
41 * unreachability from the Java side will free it.
42 */
43 jed = JNI_newObjectLocked(s_ErrorData_class, s_ErrorData_init,
44 pljava_DualState_key(), (jlong)0, p2l.longVal);
45 return jed;
46}

196 p2l.longVal = _this;
197
198 /* unpack MAKE_SQLSTATE code
199 */
200 errCode = ((ErrorData*)p2l.ptrVal)->sqlerrcode;
201 for (idx = 0; idx < 5; ++idx)
202 {
203 buf[idx] = (char)PGUNSIXBIT(errCode); /*why not cast in macro?*/
204 errCode >>= 6;
205 }
206 buf[idx] = 0;
207 result = String_createJavaStringFromNTS(buf);
208
209 END_NATIVE
