37/*
38 * LocalDate data type. This is introduced with JDBC 4.2 and Java 8. For
39 * backward-compatibility reasons it does not become the default class returned
40 * by getObject() for a PostgreSQL date, but application code in Java 8+ can and
41 * should prefer it, by passing LocalDate.class to getObject. It represents a
42 * purely local, non-zoned, notion of date, which is exactly what PostgreSQL
43 * date represents, so the correspondence is direct, with no need to fudge up
44 * some timezone info just to shoehorn the data into java.sql.Date.
45 */
46
47/*
48 * This only answers true for (same class or) DATEOID.
49 * The obtainer (below) only needs to construct and remember one instance.
50 */
51static bool _LocalDate_canReplaceType(Type self, Type other)
52{
53 TypeClass cls = Type_getClass(other);
54 return Type_getClass(self) == cls || Type_getOid(other) == DATEOID;
55}

91/*
92 * Date data type. Postgres will pass and expect number of days since
93 * Jan 01 2000. Java uses number of millisecs since midnight Jan 01 1970.
94 */
95static jvalue _Date_coerceDatum(Type self, Datum arg)
96{
97 DateADT pgDate = DatumGetDateADT(arg);
98 int64 ts = (int64)pgDate * INT64CONST(43200000000);
99 int tz = Timestamp_getTimeZone_id(ts); /* ts in 2 usec units */
100
101 jlong date = (jlong)(pgDate + EPOCH_DIFF);
102
103 jvalue result;
104 date *= 86400L; // Convert to seconds
105 date += tz; // Add local timezone
106 result.l = JNI_newObject(s_Date_class, s_Date_init, date * 1000);
107 return result;
108}

110static Datum _Date_coerceObject(Type self, jobject date)
111{
112 jlong milliSecs =
113 JNI_callLongMethod(date, s_Date_getTime)
114 - INT64CONST(86400000) * EPOCH_DIFF;
115 jlong secs =
116 milliSecs / 1000
117 - Timestamp_getTimeZone_id(milliSecs * 500); /* those 2 usec units */
118 return DateADTGetDatum((DateADT)(secs / 86400));
119}
