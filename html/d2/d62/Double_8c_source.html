46static jvalue _doubleArray_coerceDatum(Type self, Datum arg)
47{
48 jvalue result;
49 ArrayType* v = DatumGetArrayTypeP(arg);
50 jsize nElems = (jsize)ArrayGetNItems(ARR_NDIM(v), ARR_DIMS(v));
51 jdoubleArray doubleArray = JNI_newDoubleArray(nElems);
52
53 if(ARR_HASNULL(v))
54 {
55 jsize idx;
56 jboolean isCopy = JNI_FALSE;
57 bits8* nullBitMap = ARR_NULLBITMAP(v);
58 jdouble* values = (jdouble*)ARR_DATA_PTR(v);
59 jdouble* elems = JNI_getDoubleArrayElements(doubleArray, &isCopy);
60 for(idx = 0; idx < nElems; ++idx)
61 {
62 if(arrayIsNull(nullBitMap, idx))
63 elems[idx] = 0;
64 else
65 elems[idx] = *values++;
66 }
67 JNI_releaseDoubleArrayElements(doubleArray, elems, JNI_COMMIT);
68 }
69 else
70 JNI_setDoubleArrayRegion(doubleArray, 0, nElems, (jdouble*)ARR_DATA_PTR(v));
71 result.l = (jobject)doubleArray;
72 return result;
73}
