85static bool coerceObjects(void* ePlan, jobjectArray jvalues, Datum** valuesPtr, char** nullsPtr)
86{
87 char* nulls = 0;
88 Datum* values = 0;
89
90 int count = SPI_getargcount(ePlan);
91 if((jvalues == 0 && count != 0)
92 || (jvalues != 0 && count != JNI_getArrayLength(jvalues)))
93 {
94 Exception_throw(ERRCODE_PARAMETER_COUNT_MISMATCH,
95 "Number of values does not match number of arguments for prepared plan");
96 return false;
97 }
98
99 if(count > 0)
100 {
101 int idx;
102 jobject typeMap = Invocation_getTypeMap();
103 values = (Datum*)palloc(count * sizeof(Datum));
104 for(idx = 0; idx < count; ++idx)
105 {
106 Oid typeId = SPI_getargtypeid(ePlan, idx);
107 Type type = Type_fromOid(typeId, typeMap);
108 jobject value = JNI_getObjectArrayElement(jvalues, idx);
109 if(value != 0)
110 {
111 values[idx] = Type_coerceObjectBridged(type, value);
112 JNI_deleteLocalRef(value);
113 }
114 else
115 {
116 values[idx] = 0;
117 if(nulls == 0)
118 {
119 nulls = (char*)palloc(count+1);
120 memset(nulls, ' ', count); /* all values non-null initially */
121 nulls[count] = 0;
122 *nullsPtr = nulls;
123 }
124 nulls[idx] = 'n';
125 }
126 }
127 }
128 *valuesPtr = values;
129 *nullsPtr = nulls;
130 return true;
131}
