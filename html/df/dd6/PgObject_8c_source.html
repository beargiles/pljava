69char* PgObject_getClassName(jclass cls)
70{
71 jstring jstr;
72 char* tmp;
73
74 if(s_Class_getName == 0)
75 {
76 if(s_loopLock)
77 return "<exception while obtaining Class.getName()>";
78 s_loopLock = true;
79 s_Class_class = (jclass)JNI_newGlobalRef(PgObject_getJavaClass("java/lang/Class"));
80 s_Class_getName = PgObject_getJavaMethod(s_Class_class, "getName", "()Ljava/lang/String;");
81 s_loopLock = false;
82 }
83
84 jstr = (jstring)JNI_callObjectMethod(cls, s_Class_getName);
85 tmp = String_createNTS(jstr);
86 JNI_deleteLocalRef(jstr);
87 return tmp;
88}

90void PgObject_throwMemberError(jclass cls, const char* memberName, const char* signature, bool isMethod, bool isStatic)
91{
92 JNI_exceptionDescribe();
93 JNI_exceptionClear();
94 ereport(ERROR, (
95 errmsg("Unable to find%s %s %s.%s with signature %s",
96 (isStatic ? " static" : ""),
97 (isMethod ? "method" : "field"),
98 PgObject_getClassName(cls),
99 memberName,
100 signature)));
101}

103jclass PgObject_getJavaClass(const char* className)
104{
105 jclass cls = JNI_findClass(className);
106 if(cls == 0)
107 {
108 if(JNI_exceptionCheck())
109 {
110 JNI_exceptionDescribe();
111 JNI_exceptionClear();
112 }
113 ereport(ERROR, (
114 errmsg("Unable to load class %s using module path '%s'",
115 className, effectiveModulePath == 0 ? "null" :
116 effectiveModulePath)));
117 }
118 return cls;
119}

128void PgObject_registerNatives2(jclass cls, JNINativeMethod* methods)
129{
130#ifndef GCJ
131 jint nMethods = 0;
132 JNINativeMethod* m = methods;
133 while(m->name != 0)
134 {
135 m++;
136 nMethods++;
137 }
138
139 if(JNI_registerNatives(cls, methods, nMethods) != 0)
140 {
141 JNI_exceptionDescribe();
142 JNI_exceptionClear();
143 ereport(ERROR, (
144 errmsg("Unable to register native methods")));
145 }
146#endif
147}
