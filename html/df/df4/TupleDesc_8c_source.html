31/*
32 * org.postgresql.pljava.TupleDesc type.
33 * This makes a non-reference-counted copy in JavaMemoryContext of the supplied
34 * TupleDesc, which will be freed later when Java code calls the native method
35 * _free(). Therefore the caller is done with its TupleDesc when this returns.
36 */
37jobject pljava_TupleDesc_create(TupleDesc td)
38{
39 jobject jtd = 0;
40 if(td != 0)
41 {
42 MemoryContext curr = MemoryContextSwitchTo(JavaMemoryContext);
43 jtd = pljava_TupleDesc_internalCreate(td);
44 MemoryContextSwitchTo(curr);
45 }
46 return jtd;
47}

49jobject pljava_TupleDesc_internalCreate(TupleDesc td)
50{
51 jobject jtd;
52 Ptr2Long tdH;
53
54 td = CreateTupleDescCopyConstr(td);
55 tdH.longVal = 0L; /* ensure that the rest is zeroed out */
56 tdH.ptrVal = td;
57 /*
58 * Passing (jlong)0 as the ResourceOwner means this will never be matched by a
59 * nativeRelease call; that's appropriate (for now) as the TupleDesc copy is
60 * being made into JavaMemoryContext, which never gets reset, so only
61 * unreachability from the Java side will free it.
62 * XXX what about invalidating if DDL alters the column layout?
63 */
64 jtd = JNI_newObjectLocked(s_TupleDesc_class, s_TupleDesc_init,
65 pljava_DualState_key(), (jlong)0, tdH.longVal, (jint)td->natts);
66 return jtd;
67}

69/*
70 * Returns NULL if an exception has been thrown for an invalid attribute index
71 * (caller should expeditiously return), otherwise the Type for the column data
72 * (the one representing the boxing Object type, in the primitive case).
73 */
74Type pljava_TupleDesc_getColumnType(TupleDesc tupleDesc, int index)
75{
76 Type type;
77 Oid typeId = SPI_gettypeid(tupleDesc, index);
78 if(!OidIsValid(typeId))
79 {
80 Exception_throw(ERRCODE_INVALID_DESCRIPTOR_INDEX,
81 "Invalid attribute index \"%d\"", (int)index);
82 type = 0;
83 }
84 else /* Type_objectTypeFromOid returns boxed types, when that matters */
85 type = Type_objectTypeFromOid(typeId, Invocation_getTypeMap());
86 return type;
87}
