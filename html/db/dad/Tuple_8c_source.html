57jobject pljava_Tuple_internalCreate(HeapTuple ht, bool mustCopy)
58{
59 jobject jht;
60 Ptr2Long htH;
61
62 if(mustCopy)
63 ht = heap_copytuple(ht);
64
65 htH.longVal = 0L; /* ensure that the rest is zeroed out */
66 htH.ptrVal = ht;
67 /*
68 * Passing (jlong)0 as the ResourceOwner means this will never be matched by a
69 * nativeRelease call; that's appropriate (for now) as the Tuple copy is
70 * being made into JavaMemoryContext, which never gets reset, so only
71 * unreachability from the Java side will free it.
72 * XXX? this seems like a lot of tuple copying.
73 */
74 jht = JNI_newObjectLocked(s_Tuple_class, s_Tuple_init,
75 pljava_DualState_key(), (jlong)0, htH.longVal);
76 return jht;
77}

112jobject
113pljava_Tuple_getObject(
114 TupleDesc tupleDesc, HeapTuple tuple, int index, jclass rqcls)
115{
116 jobject result = 0;
117 PG_TRY();
118 {
119 Type type = pljava_TupleDesc_getColumnType(tupleDesc, index);
120 if(type != 0)
121 {
122 bool wasNull = false;
123 Datum binVal = SPI_getbinval(tuple, tupleDesc, (int)index, &wasNull);
124 if(!wasNull)
125 result = Type_coerceDatumAs(type, binVal, rqcls).l;
126 }
127 }
128 PG_CATCH();
129 {
130 Exception_throw_ERROR("SPI_getbinval");
131 }
132 PG_END_TRY();
133 return result;
134}
