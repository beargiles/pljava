386/*
387 * 'Reserve' the static parameter frame for (refArgCount,primArgCount) reference
388 * and primitive parameters, respectively, pushing temporarily out of the way
389 * any current contents, detected by a non-(0,0) existing reservation. Returns
390 * the sum of its two arguments.
391 *
392 * The corresponding pop of the earlier contents will happen at
393 * Invocation_popInvocation time, so this scheme is only appropriately used for
394 * calls that happen within the scope of an Invocation, as conventional PL/Java
395 * function calls do.
396 *
397 * It is possible to reserve (0,0) space, though no existing frame will be
398 * saved/restored in that case. Caution: the two count arguments here count only
399 * parameters, not the possibility that a primitive-returning function uses a
400 * slot in the frame for its return. The primitive call wrappers must make their
401 * own arrangements to save the typically-only-one-jvalue affected by that use
402 * and restore it on both normal and exceptional return paths. To streamline the
403 * most common case, Invocation_{push,pop}Invocation will unconditionally save
404 * the first jvalue slot, and restore it if the more heavyweight frame-pushing
405 * has not been used. That spares a primitive call wrapper the cycles of
406 * managing another PG_TRY block. Any wrapper that will use more than the first
407 * jvalue slot for returns, though, must handle its own normal and exceptional
408 * cleanup.
409 */
410static inline jsize
411reserveParameterFrame(jsize refArgCount, jsize primArgCount)
412{
413 jshort newCounts = COUNTCHECK(refArgCount, primArgCount);
414
415 /* The *s_countCheck field in the parameter area will be zero unless
416 * this is a recursive invocation (believed only possible via a UDT
417 * function called while converting the parameters for some outer
418 * invocation). It could also be zero if this is a recursive invocation
419 * but the outer one involves no parameters, which won't happen if UDT
420 * conversion for a parameter is the only way to get here, and even if
421 * it happens, we still don't need to save its frame because there is
422 * nothing there that we'll clobber.
423 */
424 if ( 0 != newCounts && 0 != *s_countCheck )
425 {
426 JNI_callStaticVoidMethodLocked(
427 s_ParameterFrame_class, s_ParameterFrame_push);
428 /* Record, in currentInvocation, that a frame was pushed; the pop
429 * will happen in Invocation_popInvocation, which our caller
430 * arranges for both normal return and PG_CATCH cases.
431 */
432 currentInvocation->frameLimits = FRAME_LIMITS_PUSHED;
433 }
434 *s_countCheck = newCounts;
435
436 return refArgCount + primArgCount;
437}

470/*
471 * Invoke an Invocable that was obtained by invoking an Invocable for a
472 * set-returning-function that returns results in value-per-call style.
473 * Pass true for 'close' when no more results are wanted. Will always overwrite
474 * *result; check the boolean return value to determine whether that is a real
475 * result (true) or the end of results was reached (false).
476 */
477jboolean pljava_Function_vpcInvoke(
478 Function self, jobject invocable, jobject rowcollect, jlong call_cntr,
479 jboolean close, jobject *result)
480{
481 /*
482 * When retrieving the very first row, this call happens under the same
483 * Invocation as the call to the user function itself that returned this
484 * invocable (and may, rarely, have pushed a ParameterFrame). What does
485 * the reservation below imply for ParameterFrame management?
486 *
487 * It's ok, because the user function's invocation will have cleared the
488 * static area parameter counts; this reservation will therefore not see a
489 * need to push a frame. If one was pushed for the user function itself, it
490 * remains on top, to be popped when the Invocation is.
491 */
492 reserveParameterFrame(1, 2);
493
494 JNI_setObjectArrayElement(s_referenceParameters, 0, rowcollect);
495 s_primitiveParameters[0].j = call_cntr;
496 s_primitiveParameters[1].z = close;
497 *result = JNI_callStaticObjectMethod(s_EntryPoints_class,
498 s_EntryPoints_invoke, invocable);
499 return s_primitiveParameters[0].z;
500}

567/*
568 * This checks specifically for a "Java-based scalar" a/k/a BaseUDT,
569 * and will call UDT_registerUDT accordingly if it is.
570 */
571Type Function_checkTypeBaseUDT(Oid typeId, Form_pg_type typeStruct)
572{
573 HeapTuple procTup;
574 Datum d;
575 Form_pg_proc procStruct;
576 Type t = NULL;
577 jstring schemaName;
578 jclass clazz = NULL;
579 jclass t_clazz = NULL;
580
581 Oid procId[4] =
582 {
583 typeStruct->typinput, typeStruct->typreceive,
584 typeStruct->typsend, typeStruct->typoutput
585 };
586 jmethodID getter[4] =
587 {
588 s_Function_udtParseHandle, s_Function_udtReadHandle,
589 s_Function_udtWriteHandle, s_Function_udtToStringHandle
590 };
591 char *langName[4] = { NULL, NULL, NULL, NULL };
592 bool trusted[4];
593 jobject handle[4] = { NULL, NULL, NULL, NULL };
594 int i;
595
596 for ( i = 0; i < 4; ++ i )
597 {
598 if ( ! InstallHelper_isPLJavaFunction(
599 procId[i], &langName[i], &trusted[i]) )
600 break;
601 }
602
603 /*
604 * If that loop did not find all four support functions to be PL/Java ones,
605 * we have struck out; pfree anything it did find and return the bad news.
606 */
607 if ( i < 4 )
608 {
609 for ( ; i >= 0 ; -- i )
610 if ( NULL != langName[i] )
611 pfree(langName[i]);
612 return NULL;
613 }
614
615 /*
616 * At this point, it is looking like a PL/Java BaseUDT; we have the four
617 * support function oids and the language names and trusted flags to go
618 * with them.
619 *
620 * Must still confirm that (1) each one is declared with a UDT[classname]
621 * style of AS string (getClassIfUDT returns null if not), (2) the named
622 * class inherits from SQLData (ClassCastException happens if not), and
623 * (3) that's the same class for all four of them (bail from this loop
624 * and goto classMismatch if not). We'll also consider it a classMismatch
625 * if some but not all getClassIfUDT results are null.
626 *
627 * Provided none of that goes wrong, obtain their handles in this loop too.
628 */
629 for ( i = 0; i < 4; ++ i )
630 {
631 /*
632 * Get the pg_proc info corresponding to support function i,
633 * needed by getClassIfUDT().
634 */
635 procTup = PgObject_getValidTuple(PROCOID, procId[i], "function");
636 procStruct = (Form_pg_proc)GETSTRUCT(procTup);
637 schemaName = getSchemaName(procStruct->pronamespace);
638 d = heap_copy_tuple_as_datum(
639 procTup, Type_getTupleDesc(s_pgproc_Type, 0));
640
641 t_clazz = (jclass)JNI_callStaticObjectMethod(s_Function_class,
642 s_Function_getClassIfUDT, Type_coerceDatum(s_pgproc_Type, d),
643 schemaName);
644
645 pfree((void *)d);
646 JNI_deleteLocalRef(schemaName);
647 ReleaseSysCache(procTup);
648
649 /*
650 * Save the first clazz returned; for subsequent ones, just confirm it's
651 * not different, then delete the extra local ref.
652 */
653 if ( 0 == i )
654 clazz = t_clazz;
655 else
656 {
657 if ( JNI_FALSE == JNI_isSameObject(clazz, t_clazz) )
658 goto classMismatch;
659 JNI_deleteLocalRef(t_clazz);
660 }
661
662 if ( NULL == clazz )
663 continue;
664
665 handle[i] = obtainUDTHandle(getter[i], clazz, langName[i], trusted[i]);
666 }
667
668 /*
669 * We can only be here if getClassIfUDT returned the same value for clazz
670 * all four times. But that value could have been NULL; no UDT if so.
671 */
672 if ( NULL != clazz )
673 t = (Type)UDT_registerUDT(clazz, typeId, typeStruct, 0, true,
674 handle[0], handle[1], handle[2], handle[3]);
675 /*
676 * UDT_registerUDT will already have called JNI_deleteLocalRef on the
677 * four handles. (Or clazz was NULL and there aren't any to delete anyway.)
678 */
679 JNI_deleteLocalRef(clazz);
680 for ( i = 0; i < 4; ++ i )
681 pfree(langName[i]);
682
683 return t;
684
685classMismatch:
686 while ( i --> 0 )
687 JNI_deleteLocalRef(handle[i]);
688 for ( i = 0; i < 4; ++ i )
689 pfree(langName[i]);
690 JNI_deleteLocalRef(clazz);
691 JNI_deleteLocalRef(t_clazz);
692 ereport(ERROR, (errmsg(

698 Oid funcOid, bool trusted, bool forTrigger,
699 bool forValidator, bool checkBody)
700{
701 Function self;
702 HeapTuple procTup =
703 PgObject_getValidTuple(PROCOID, funcOid, "function");
704 Form_pg_proc procStruct = (Form_pg_proc)GETSTRUCT(procTup);
705 HeapTuple lngTup =
706 PgObject_getValidTuple(LANGOID, procStruct->prolang, "language");

709 bool ltrust = lngStruct->lanpltrusted;
710 jstring schemaName;
711 Ptr2Long p2l;

713 jobject invocable;
714
715 if ( trusted != ltrust )
716 elog(ERROR,
717 "function with oid %u invoked through wrong call handler "
718 "for %strusted language %s", funcOid, ltrust ? "" : "un",
719 NameStr(lngStruct->lanname));

722

724

726 (Function)PgObjectClass_allocInstance(s_FunctionClass,TopMemoryContext);
727 p2l.longVal = 0;
728 p2l.ptrVal = (void *)self;
729
730 PG_TRY();

732 invocable =
733 JNI_callStaticObjectMethod(s_Function_class, s_Function_create,
734 p2l.longVal, Type_coerceDatum(s_pgproc_Type, d), lname,
735 schemaName,
736 trusted ? JNI_TRUE : JNI_FALSE,
737 forTrigger ? JNI_TRUE : JNI_FALSE,
738 forValidator ? JNI_TRUE : JNI_FALSE,
739 checkBody ? JNI_TRUE : JNI_FALSE);
740 }
741 PG_CATCH();

743 JNI_deleteLocalRef(schemaName);
744 ReleaseSysCache(lngTup);
745 ReleaseSysCache(procTup);
746 pfree(self); /* would otherwise leak into TopMemoryContext */

748 }

755 /*
756 * One of four things has happened, the product of two binary choices:
757 * - This Function turns out to be either a UDT function, or a nonUDT one.
758 * - it is now fully initialized and should be returned, or it isn't, and
759 * should be pfree()d. (Validator calls don't have to do the whole job.)
760 *
761 * If Function.create returned a non-NULL result, this is a fully
762 * initialized, non-UDT function, ready to save and use. (That can happen
763 * even during validation; if checkBody is true, enough work is done to get
764 * a complete result, so we might as well save it.)
765 *
766 * If it returned NULL, this is either an incompletely-initialized non-UDT
767 * function, or it is a UDT function (whether fully initialized or not; it
768 * is always NULL for a UDT function). If it is a UDT function and not
769 * complete, it should be pfree()d. If complete, it has already been
770 * registered with the UDT machinery and should be saved. We can arrange
771 * (see _storeToUDT below) for the isUDT flag to be left false if the UDT
772 * initialization isn't complete; that collapses the need-to-pfree cases
773 * into one case here (Function.create returned NULL && ! isUDT).
774 *
775 * Because allocInstance zeroes memory, isUDT is reliably false even if
776 * the Java code bailed early.
777 */
778
779 if ( NULL != invocable )
780 {
781 self->func.nonudt.invocable = JNI_newGlobalRef(invocable);
782 JNI_deleteLocalRef(invocable);
783 }
784 else if ( ! self->isUDT )
785 {
786 pfree(self);
787 if ( forValidator )
788 return NULL;
789 elog(ERROR,
790 "failed to create a PL/Java function (oid %u) and not validating",
791 funcOid);
792 }
793
794 return self;
795}

899Datum
900Function_invoke(
901 Oid funcoid, bool trusted, bool forTrigger, bool forValidator,
902 bool checkBody, PG_FUNCTION_ARGS)
903{
904 Function self;
905 Datum retVal;
906 Size passedArgCount;
907 Type invokerType;
908 bool skipParameterConversion = false;
909
910 self = getFunction(funcoid, trusted, forTrigger, forValidator, checkBody);
911
912 if ( forValidator )
913 PG_RETURN_VOID();
914
915 if ( forTrigger )
916 return invokeTrigger(self, fcinfo);
917
918 fcinfo->isnull = false;
919
920 if(self->isUDT)
921 return self->func.udt.udtFunction(self->func.udt.udt, fcinfo);
922
923 if ( self->func.nonudt.isMultiCall )
924 {
925 if ( SRF_IS_FIRSTCALL() )
926 {
927 /* A class loader or other mechanism might have connected already.
928 * This connection must be dropped since its parent context
929 * is wrong.
930 */
931 Invocation_assertDisconnect();
932 }
933 else
934 {
935 /* In PL/Java's implementation of the ValuePerCall SRF protocol, the
936 * passed parameters from SQL only matter on the first call. All
937 * subsequent calls are either hasNext()/next() on an Iterator, or
938 * assignRowValues on a ResultSetProvider, and none of those methods
939 * will receive the SQL-passed parameters. So there is no need to
940 * spend cycles to convert them and populate the parameter area.
941 */
942 skipParameterConversion = true;
943 }
944 }
945
946 passedArgCount = PG_NARGS();
947
948 if ( ! skipParameterConversion )
949 {
950 jsize reservedArgCount = reserveParameterFrame(
951 self->func.nonudt.numRefParams, self->func.nonudt.numPrimParams);
952
953 if ( passedArgCount != reservedArgCount
954 && passedArgCount + 1 != reservedArgCount ) /* the OUT arg case */
955 elog(ERROR, "function expecting %u arguments passed %u",
956 (unsigned int)reservedArgCount, (unsigned int)passedArgCount);
957 }
958
959 installContextLoader(self);
960
961 invokerType = self->func.nonudt.returnType;
962
963 if ( passedArgCount > 0 && ! skipParameterConversion )
964 {
965 int32 idx;
966 int32 refIdx = 0;
967 int32 primIdx = 0;
968 Type* types = self->func.nonudt.paramTypes;
969 jvalue coerced;
970
971 if(Type_isDynamic(invokerType))
972 invokerType = Type_getRealType(invokerType,
973 get_fn_expr_rettype(fcinfo->flinfo), self->func.nonudt.typeMap);
974
975 for(idx = 0; idx < passedArgCount; ++idx)
976 {
977 Type paramType = types[idx];
978 bool passPrimitive = passAsPrimitive(paramType);
979
980 if(PG_ARGISNULL(idx))
981 {
982 /*
983 * Set this argument to zero (or null in case of object)
984 */
985 if ( passPrimitive )
986 s_primitiveParameters[primIdx++].j = 0L;
987 else
988 ++ refIdx; /* array element is already initially null */
989 }
990 else
991 {
992 if(Type_isDynamic(paramType))
993 paramType = Type_getRealType(paramType,
994 get_fn_expr_argtype(fcinfo->flinfo, idx),
995 self->func.nonudt.typeMap);
996 coerced = Type_coerceDatum(paramType, PG_GETARG_DATUM(idx));
997 if ( passPrimitive )
998 s_primitiveParameters[primIdx++] = coerced;
999 else
1000 JNI_setObjectArrayElement(
1001 s_referenceParameters, refIdx++, coerced.l);
1002 }
1003 }
1004 }
1005
1006 retVal = self->func.nonudt.isMultiCall
1007 ? Type_invokeSRF(invokerType, self, fcinfo)
1008 : Type_invoke(invokerType, self, fcinfo);
1009
1010 return retVal;
1011}

1013/*
1014 * Invoke a trigger. Wrap the TriggerData in org.postgresql.pljava.TriggerData
1015 * object, make the call, and unwrap the resulting Tuple.
1016 */
1017static inline Datum
1018invokeTrigger(Function self, PG_FUNCTION_ARGS)
1019{
1020 jobject jtd;
1021 Datum ret;
1022
1023 TriggerData *td = (TriggerData*)fcinfo->context;
1024 jtd = pljava_TriggerData_create(td);
1025 if(jtd == 0)
1026 return 0;
1027
1028 reserveParameterFrame(1, 0);
1029 installContextLoader(self);
1030
1031 JNI_setObjectArrayElement(s_referenceParameters, 0, jtd);
1032
1033#if PG_VERSION_NUM >= 100000
1034 currentInvocation->triggerData = td;
1035 /* Also starting in PG 10, Invocation_assertConnect must be called before
1036 * the getTriggerReturnTuple below. That could be done right here, but at
1037 * the risk of changing the memory context from what the invoked trigger
1038 * function expects. More cautiously, add the assertConnect later, after
1039 * the trigger function has returned.
1040 */
1041#endif
1042 Type_invoke(self->func.nonudt.returnType, self, fcinfo);
1043
1044 fcinfo->isnull = false;
1045 if(JNI_exceptionCheck())
1046 ret = 0;
1047 else
1048 {
1049 /* A new Tuple may or may not be created here. Ensure that, if it is,
1050 * it is created in the upper context (even after connecting SPI, should
1051 * that be necessary).
1052 */
1053 MemoryContext currCtx;
1054#if PG_VERSION_NUM >= 100000
1055 /* If the invoked trigger function didn't connect SPI, do that here
1056 * (getTriggerReturnTuple now needs it), but there will be no need to
1057 * register the triggerData in that case.
1058 */
1059 currentInvocation->triggerData = NULL;
1060 Invocation_assertConnect();
1061#endif
1062 currCtx = Invocation_switchToUpperContext();

1064 pljava_TriggerData_getTriggerReturnTuple(
1065 jtd, &fcinfo->isnull));
1066
1067 /* Triggers are not allowed to set the fcinfo->isnull, even when
1068 * they return null.
1069 */
1070 fcinfo->isnull = false;
1071
1072 MemoryContextSwitchTo(currCtx);
1073 }
1074
1075 JNI_deleteLocalRef(jtd);
1076 return ret;
1077}
