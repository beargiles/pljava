19jobject SQLOutputToChunk_create(StringInfo data, bool isJavaBasedScalar)
20{
21 jobject dbb;
22 Ptr2Long p2l;
23 p2l.longVal = 0L; /* ensure that the rest is zeroed out */
24 p2l.ptrVal = data;
25 dbb = JNI_newDirectByteBuffer(data->data, data->maxlen);
26 if ( 0 < data->len )
27 JNI_callObjectMethodLocked(dbb, s_Buffer_position, data->len);
28 return JNI_newObject(s_SQLOutputToChunk_class, s_SQLOutputToChunk_init,
29 p2l.longVal, dbb, isJavaBasedScalar ? JNI_TRUE : JNI_FALSE);

32void SQLOutputToChunk_close(jobject stream)
33{
34 /*
35 * The close method calls ensureCapacity(0), so thanks to _ensureCapacity
36 * below, on return the StringInfo len is correct and the contents are
37 * NUL terminated, (re-)establishing the StringInfo invariant.
38 */
39 JNI_callVoidMethod(stream, s_SQLOutputToChunk_close);

93 /*
94 * The StringInfo functions maintain an invariant that the contents are
95 * NUL-terminated. That is *not* assured in general while Java pokes at it
96 * via the ByteBuffer, but is restored here at every call to _ensureCapacity

98 * is always arranged by enlargeStringInfo, there is room for this even if
99 * zero was passed for 'needed', as happens when closing.
100 */

103 goto done;
