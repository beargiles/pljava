17/* The byte maps to the postgres type "char", i.e. the
18 * 8 bit, one byte quantity. The java byte was chosen instead of

21 */

43static jvalue _byteArray_coerceDatum(Type self, Datum arg)
44{
45 jvalue result;
46 ArrayType* v = DatumGetArrayTypeP(arg);
47 jsize nElems = (jsize)ArrayGetNItems(ARR_NDIM(v), ARR_DIMS(v));
48 jbyteArray byteArray = JNI_newByteArray(nElems);
49
50 if(ARR_HASNULL(v))
51 {
52 jsize idx;
53 jboolean isCopy = JNI_FALSE;
54 bits8* nullBitMap = ARR_NULLBITMAP(v);
55 jbyte* values = (jbyte*)ARR_DATA_PTR(v);
56 jbyte* elems = JNI_getByteArrayElements(byteArray, &isCopy);
57 for(idx = 0; idx < nElems; ++idx)
58 {
59 if(arrayIsNull(nullBitMap, idx))
60 elems[idx] = 0;
61 else
62 elems[idx] = *values++;
63 }
64 JNI_releaseByteArrayElements(byteArray, elems, JNI_COMMIT);
65 }
66 else
67 JNI_setByteArrayRegion(byteArray, 0, nElems, (jbyte*)ARR_DATA_PTR(v));
68 result.l = (jobject)byteArray;
69 return result;
70}

72static Datum _byteArray_coerceObject(Type self, jobject byteArray)
73{
74 ArrayType* v;
75 jsize nElems;
76
77 if(byteArray == 0)
78 return 0;
79
80 nElems = JNI_getArrayLength((jarray)byteArray);
81 v = createArrayType(nElems, sizeof(jbyte), CHAROID, false);
82 JNI_getByteArrayRegion((jbyteArray)byteArray, 0, nElems, (jbyte*)ARR_DATA_PTR(v));
83
84 PG_RETURN_ARRAYTYPE_P(v);
85}

116void Byte_initialize(void)
117{
118 Type t_byte;
119 Type t_Byte;
120 TypeClass cls;
121
122 s_Byte_class = JNI_newGlobalRef(PgObject_getJavaClass("java/lang/Byte"));
123 s_Byte_init = PgObject_getJavaMethod(s_Byte_class, "<init>", "(B)V");
124 s_Byte_byteValue = PgObject_getJavaMethod(s_Byte_class, "byteValue", "()B");
125
126 cls = TypeClass_alloc("type.Byte");
127 cls->canReplaceType = _Byte_canReplace;
128 cls->JNISignature = "Ljava/lang/Byte;";
129 cls->javaTypeName = "java.lang.Byte";
130 cls->coerceDatum = _Byte_coerceDatum;
131 cls->coerceObject = _Byte_coerceObject;
132 t_Byte = TypeClass_allocInstance(cls, CHAROID);
133
134 cls = TypeClass_alloc("type.byte");
135 cls->JNISignature = "B";
136 cls->javaTypeName = "byte";
137 cls->invoke = _byte_invoke;
138 cls->coerceDatum = _byte_coerceDatum;
139 cls->coerceObject = _Byte_coerceObject;
140 cls->createArrayType = _byte_createArrayType;
141 s_byteClass = cls;
142
143 t_byte = TypeClass_allocInstance(cls, CHAROID);
144 t_byte->objectType = t_Byte;
145 Type_registerType("byte", t_byte);
146 Type_registerType("java.lang.Byte", t_Byte);
147}
