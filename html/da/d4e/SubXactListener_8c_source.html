23static void subXactCB(
24 SubXactEvent event, SubTransactionId mySubid, SubTransactionId parentSubid,
25 void* arg)
26{
27 /*
28 * Map the subids to PgSavepoints first - this function upcalls into Java
29 * without releasing the Backend.THREADLOCK monitor, so the called methods
30 * can know they're on the PG thread; Backend.threadMayEnterPG() is true.
31 * It is important to look up mySubid before parentSubid, as it is possible
32 * a new PgSavepoint instance is under construction in the 'nursery', and
33 * will be assigned the first id to be looked up.
34 */
35 jobject sp = pljava_PgSavepoint_forId(mySubid);
36 jobject parent = pljava_PgSavepoint_forId(parentSubid);
37
38 /*
39 * These upcalls are made with the monitor released. We are, of course, ON
40 * the PG thread, but this time with no monitor held to prevent another
41 * thread from stepping in. These methods should not blindly assert
42 * Backend.threadMayEnterPG(), as for some java_thread_pg_entry settings it
43 * won't be true. This is the legacy behavior, so not changed for 1.5.x.
44 *
45 * The event ordinal can simply be passed to Java, as long as upstream
46 * hasn't changed the order; list the known ones in a switch, for a better
47 * chance that a clever compiler will warn if upstream has added any.
48 */
49 switch(event)
50 {
51 case SUBXACT_EVENT_START_SUB:
52 case SUBXACT_EVENT_COMMIT_SUB:
53 case SUBXACT_EVENT_ABORT_SUB:
54 case SUBXACT_EVENT_PRE_COMMIT_SUB:
55 JNI_callStaticVoidMethod(s_SubXactListener_class,
56 s_SubXactListener_invokeListeners, (jint)event, sp, parent);
57 }
58}
