59/*
60 * For VarlenaWrapper.Output, define a dead-simple "expanded object" format
61 * consisting of linked allocated blocks, so if a long value is being written,
62 * it does not have to get repeatedly reallocated and copied. The "expanded
63 * object" form is a valid sort of PostgreSQL Datum, and can be passed around
64 * in that form, and reparented between memory contexts with different
65 * lifetimes; when a time comes that PostgreSQL needs it in a 'flattened'
66 * form, it will use these 'methods' to flatten it, and that's when the one
67 * final reallocation and copy will happen.
68 */
69
70static Size VOS_get_flat_size(ExpandedObjectHeader *eohptr);
71static void VOS_flatten_into(ExpandedObjectHeader *eohptr,
72 void *result, Size allocated_size);
73
74static const ExpandedObjectMethods VOS_methods =
75{
76 VOS_get_flat_size,
77 VOS_flatten_into
78};

97/*
98 * Create and return a VarlenaWrapper.Input allowing Java to read the content
99 * of an existing Datum d, which must be a varlena type (assumed, not checked
100 * here).
101 *
102 * The datum will be copied (detoasting if need be) into a memory context with
103 * parent as its parent, so it can be efficiently reparented later if adopted,
104 * and the VarlenaWrapper will be associated with the ResourceOwner ro, which
105 * determines its lifespan (if not adopted). The ResourceOwner needs to be one
106 * that will be released no later than the memory context itself.
107 */
108jobject pljava_VarlenaWrapper_Input(
109 Datum d, MemoryContext parent, ResourceOwner ro)
110{
111 jobject vr;
112 jobject dbb;
113 MemoryContext mc;
114 MemoryContext prevcxt;
115 _VL_TYPE vl;
116 Ptr2Long p2lro;
117 Ptr2Long p2lcxt;
118 Ptr2Long p2lpin;
119 Ptr2Long p2ldatum;
120 Size parked;
121 Size actual;
122 Snapshot pin = NULL;
123
124 vl = (_VL_TYPE) DatumGetPointer(d);
125
126 if ( VARATT_IS_EXTERNAL_INDIRECT(vl) ) /* at most once; can't be nested */
127 {
128 struct varatt_indirect redirect;
129 VARATT_EXTERNAL_GET_POINTER(redirect, vl);
130 vl = (_VL_TYPE)redirect.pointer;
131 d = PointerGetDatum(vl);
132 }
133
134 parked = VARSIZE_ANY(vl);
135 actual = toast_raw_datum_size(d) - VARHDRSZ;
136
137 mc = AllocSetContextCreate(parent, "PL/Java VarlenaWrapper.Input",
138 ALLOCSET_START_SMALL_SIZES);
139
140 prevcxt = MemoryContextSwitchTo(mc);
141
142 if ( actual < 4096 || (actual >> 1) < parked )
143 goto justDetoastEagerly;
144 if ( VARATT_IS_EXTERNAL_EXPANDED(vl) )
145 goto justDetoastEagerly;
146 if ( VARATT_IS_EXTERNAL_ONDISK(vl) )
147 {
148 pin = GetOldestSnapshot();
149 if ( NULL == pin )
150 {
151 /*
152 * Unable to register a snapshot and just park the tiny pointer.
153 * If it points to compressed data, can still park that rather than
154 * fully detoasting.
155 */
156 struct varatt_external toast_pointer;
157 VARATT_EXTERNAL_GET_POINTER(toast_pointer, vl);
158 parked = VARATT_EXTERNAL_GET_EXTSIZE(toast_pointer) + VARHDRSZ;
159 if ( (actual >> 1) < parked ) /* not compressed enough to bother */
160 goto justDetoastEagerly;
161 vl = detoast_external_attr(vl); /* fetch without decompressing */
162 d = PointerGetDatum(vl);
163 dbb = NULL;
164 goto constructResult;
165 }
166 pin = RegisterSnapshotOnOwner(pin, ro);
167 }
168
169/* parkAndDetoastLazily: */
170 vl = (_VL_TYPE) DatumGetPointer(datumCopy(d, false, -1));
171 dbb = NULL;
172 goto constructResult;
173
174justDetoastEagerly:
175 vl = (_VL_TYPE) PG_DETOAST_DATUM_COPY(d);
176 parked = actual + VARHDRSZ;
177 dbb = JNI_newDirectByteBuffer(VARDATA(vl), actual);
178
179constructResult:
180 MemoryContextSwitchTo(prevcxt);
181
182 p2lro.longVal = 0L;
183 p2lcxt.longVal = 0L;
184 p2lpin.longVal = 0L;
185 p2ldatum.longVal = 0L;
186
187 p2lro.ptrVal = ro;
188 p2lcxt.ptrVal = mc;
189 p2lpin.ptrVal = pin;
190 p2ldatum.ptrVal = vl;
191
192 vr = JNI_newObjectLocked(s_VarlenaWrapper_Input_class,
193 s_VarlenaWrapper_Input_init, pljava_DualState_key(),
194 p2lro.longVal, p2lcxt.longVal, p2lpin.longVal, p2ldatum.longVal,
195 (jlong)parked, (jlong)actual, dbb);
196
197 if ( NULL != dbb )
198 JNI_deleteLocalRef(dbb);
199
200 return vr;
201}

203/*
204 * Create and return a VarlenaWrapper.Output, initially empty, into which Java
205 * can write.
206 *
207 * The datum will be assembled in the memory context mc, and the VarlenaWrapper
208 * will be associated with the ResourceOwner ro, which determines its lifespan.
209 * The ResourceOwner needs to be one that will be released no later than
210 * the memory context itself.
211 *
212 * After Java has written the content, native code can obtain the Datum by
213 * calling pljava_VarlenaWrapper_Output_adopt().
214 */
215jobject pljava_VarlenaWrapper_Output(MemoryContext parent, ResourceOwner ro)
216{
217 ExpandedVarlenaOutputStreamHeader *evosh;
218 jobject vos;
219 jobject dbb;
220 MemoryContext mc;
221 Ptr2Long p2lro;
222 Ptr2Long p2lcxt;
223 Ptr2Long p2ldatum;
224
225 mc = AllocSetContextCreate(parent, "PL/Java VarlenaWrapper.Output",
226 ALLOCSET_START_SMALL_SIZES);
227 /*
228 * Allocate an initial chunk sized to contain the expanded V.O.S. header,
229 * plus the header and data for one node to hold INITIALSIZE data bytes.
230 */
231 evosh = MemoryContextAlloc(mc,
232 sizeof *evosh + sizeof *(evosh->tail) + INITIALSIZE);
233 /*
234 * Initialize the expanded object header and its pointer to the first node.
235 */
236 EOH_init_header(&(evosh->hdr), &VOS_methods, mc);
237 evosh->total_size = VARHDRSZ;
238 evosh->tail = (ExpandedVarlenaOutputStreamNode *)(evosh + 1);
239 /*
240 * Initialize that first node.
241 */
242 evosh->tail->next = evosh->tail;
243 /* evosh->tail->size will be filled in by _nextBuffer() later */
244
245 p2lro.longVal = 0L;
246 p2lcxt.longVal = 0L;
247 p2ldatum.longVal = 0L;
248
249 p2lro.ptrVal = ro;
250 p2lcxt.ptrVal = mc;
251 p2ldatum.ptrVal = DatumGetPointer(EOHPGetRWDatum(&(evosh->hdr)));
252
253 /*
254 * The data bytes begin right after the node header struct.
255 */
256 dbb = JNI_newDirectByteBuffer(evosh->tail + 1, INITIALSIZE);
257
258 vos = JNI_newObjectLocked(s_VarlenaWrapper_Output_class,
259 s_VarlenaWrapper_Output_init, pljava_DualState_key(),
260 p2lro.longVal, p2lcxt.longVal, p2ldatum.longVal, dbb);
261 JNI_deleteLocalRef(dbb);
262
263 return vos;
264}

266/*
267 * Adopt a VarlenaWrapper (if Output, after Java code has written and closed it)
268 * and leave it no longer accessible from Java. It may be an 'expanded' datum,
269 * in PG 9.5+ where there are such things. Otherwise, it will be an ordinary
270 * flat one (the ersatz 'expanded' form used internally here then being only an
271 * implementation detail, not exposed to the caller); its memory context is
272 * unchanged.
273 */
274Datum pljava_VarlenaWrapper_adopt(jobject vlw)
275{
276 Ptr2Long p2l;
277
278 p2l.longVal = JNI_callLongMethodLocked(vlw, s_VarlenaWrapper_adopt,
279 pljava_DualState_key());
280
281 return PointerGetDatum(p2l.ptrVal);
282}

291static void VOS_flatten_into(ExpandedObjectHeader *eohptr,
292 void *result, Size allocated_size)
293{
294 ExpandedVarlenaOutputStreamHeader *evosh =
295 (ExpandedVarlenaOutputStreamHeader *)eohptr;
296 ExpandedVarlenaOutputStreamNode *node = evosh->tail;
297
298 Assert(allocated_size == evosh->total_size);
299 SET_VARSIZE(result, allocated_size);
300 result = VARDATA(result);
301
302 do
303 {
304 node = node->next;
305 memcpy(result, node + 1, node->size);
306 result = (char *)result + node->size;
307 }
308 while ( node != evosh->tail );
309}

451/*
452 * Class: org_postgresql_pljava_internal_VarlenaWrapper_Input_State
453 * Method: _fetch
454 * Signature: (JJ)J
455 *
456 * Assumption: this is only called when a snapshot has been registered (meaning
457 * the varlena is EXTERNAL_ONDISK) and the snapshot is soon to be unregistered.
458 * All that's needed is to 'fetch' the representation from disk, in case the
459 * toast rows could be subject to vacuuming after the snapshot is unregistered.
460 * A fetch is not a full detoast; if what's fetched is compressed, it stays
461 * compressed. This method does not need to unregister the snapshot, as that
462 * will happen soon anyway. It does pfree the toast pointer.
463 */
464JNIEXPORT jlong JNICALL Java_org_postgresql_pljava_internal_VarlenaWrapper_00024Input_00024State__1fetch
465 (JNIEnv *env, jobject _this, jlong varlena, jlong memContext)
466{
467 Ptr2Long p2lvl;
468 Ptr2Long p2lcxt;
469 MemoryContext prevcxt;
470 _VL_TYPE fetched;
471
472 p2lvl.longVal = varlena;
473 p2lcxt.longVal = memContext;
474
475 BEGIN_NATIVE_NO_ERRCHECK;
476 prevcxt = MemoryContextSwitchTo((MemoryContext) p2lcxt.ptrVal);
477 fetched = detoast_external_attr((_VL_TYPE) p2lvl.ptrVal);
478 pfree(p2lvl.ptrVal);
479 p2lvl.longVal = 0L;
480 p2lvl.ptrVal = fetched;
481 MemoryContextSwitchTo(prevcxt);
482 END_NATIVE;
