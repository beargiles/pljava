65/*
66 * Structure used to retain state of set-returning functions using the
67 * SFRM_ValuePerCall protocol (the only one PL/Java currently supports). In that
68 * protocol, PostgreSQL will make repeated calls arriving at Type_invokeSRF
69 * below, which returns one result row on each call (and then a no-more-results
70 * result). This struct holds necessary context through the sequence of calls.
71 *
72 * If PostgreSQL is satisfied before the whole set has been returned, the
73 * _endOfSetCB below will be invoked to clean up the work in progress, and also
74 * needs this stashed information.
75 */
76typedef struct
77{
78 Type elemType;
79 Function fn;
80 jobject rowProducer;
81 jobject rowCollector;
82 /*
83 * Invocation instance, if any, the Java counterpart to currentInvocation
84 * the C struct. There isn't one unless it gets asked for, then if it is,
85 * it's saved here, so even though the C currentInvocation really is new on
86 * each entry from PG, Java will see one Invocation instance throughout the
87 * sequence of calls.
88 */
89 jobject invocation;
90 /*
91 * Two pieces of state from Invocation.c's management of SPI connection,
92 * effectively keeping one such connection alive through the sequence of
93 * calls. I could easily be led to question the advisability of even doing
94 * that, but it has a long history in PL/Java, so changing it might call for
95 * some careful analysis.
96 */
97 MemoryContext spiContext;
98 bool hasConnected;
99} CallContextData;
100
101/*
102 * Called during evaluation of a set-returning function, at various points after
103 * calls into Java code could have instantiated an Invocation, or connected SPI.
104 * Does not stash elemType, rowProducer, or rowCollector; those are all
105 * unconditionally set in the first-call initialization, and spiContext to zero.
106 */
107static void stashCallContext(CallContextData *ctxData)
108{
109 bool wasConnected = ctxData->hasConnected;
110
111 ctxData->hasConnected = currentInvocation->hasConnected;
112
113 ctxData->invocation = currentInvocation->invocation;
114
115 if ( wasConnected )
116 return;
117
118 /*
119 * If SPI has been connected for the first time, capture the memory context
120 * it imposed. Curiously, this is not used again except in _closeIteration.
121 */
122 if(ctxData->hasConnected)
123 ctxData->spiContext = CurrentMemoryContext;
124}
125
126/*
127 * Called either at normal completion of a set-returning function, or by the
128 * _endOfSetCB if PostgreSQL doesn't want all the results.
129 */
130static void _closeIteration(CallContextData* ctxData)
131{
132 jobject dummy;
133 currentInvocation->hasConnected = ctxData->hasConnected;
134 currentInvocation->invocation = ctxData->invocation;
135
136 /*
137 * Why pass 1 as the call_cntr? We won't always have the actual call_cntr
138 * value at _closeIteration time (the _endOfSetCB isn't passed it), and the
139 * Java interfaces being used don't need it (close() isn't passed a row
140 * number), but at least 1 is different from zero, in case vpcInvoke has
141 * a reason to distinguish the first call (in the same invocation as the
142 * overall setup) from subsequent ones.
143 */
144 pljava_Function_vpcInvoke(
145 ctxData->fn, ctxData->rowProducer, NULL, 1, JNI_TRUE, &dummy);
146
147 JNI_deleteGlobalRef(ctxData->rowProducer);
148 if(ctxData->rowCollector != 0)
149 JNI_deleteGlobalRef(ctxData->rowCollector);
150
151 if(ctxData->hasConnected && ctxData->spiContext != 0)
152 {
153 /*
154 * SPI was connected. We will (1) switch back to the memory context that
155 * was imposed by SPI_connect, then (2) disconnect. SPI_finish will have
156 * switched back to whatever memory context was current when SPI_connect
157 * was called, and that context had better still be valid. It might be
158 * the executor's multi_call_memory_ctx, if the SPI_connect happened
159 * during initialization of the rowProducer or rowCollector, or the
160 * executor's per-row context, if it happened later. Both of those are
161 * still valid at this point. The final step (3) is to switch back to
162 * the context we had before (1) and (2) happened.
163 */
164 MemoryContext currCtx = MemoryContextSwitchTo(ctxData->spiContext);
165 Invocation_assertDisconnect();
166 MemoryContextSwitchTo(currCtx);
167 }
168}

170/*
171 * Called by PostgreSQL if abandoning the collection of set-returning-function
172 * results early.
173 */
174static void _endOfSetCB(Datum arg)
175{
176 Invocation ctx;
177 CallContextData* ctxData = (CallContextData*)DatumGetPointer(arg);
178
179 /*
180 * Even if there is an invocation already on the stack, there is no
181 * convincing reason to think this callback belongs to it; PostgreSQL
182 * will make this callback when the expression context we did belong to
183 * is being torn down. This is not a hot operation; it only happens in
184 * rare cases when an SRF has been called and not completely consumed.
185 * So just unconditionally set up a context for this call, and clean up
186 * our own mess.
187 */
188 PG_TRY();
189 {
190 Invocation_pushInvocation(&ctx);
191 currentInvocation->inExprContextCB = true;
192 _closeIteration(ctxData);
193 Invocation_popInvocation(false);
194 }
195 PG_CATCH();
196 {
197 Invocation_popInvocation(true);
198 PG_RE_THROW();
199 }
200 PG_END_TRY();
201}

221static Type _getCoerce(Type self, Type other, Oid fromOid, Oid toOid,
222 HashMap *map, Type builder(Type, Type, Oid))
223{
224 Oid funcId;
225 Type coercer;
226 CoercionPathType cpt;
227
228 if(*map != 0)
229 {
230 coercer = HashMap_getByOid(*map, other->typeId);
231 if(coercer != 0)
232 return coercer;
233 }
234
235 cpt = find_coercion_pathway(toOid, fromOid, COERCION_EXPLICIT, &funcId);
236 switch ( cpt )
237 {
238 case COERCION_PATH_NONE:
239 elog(ERROR, "no conversion function from (regtype) %d to %d",
240 fromOid, toOid);
241 pg_unreachable(); /*elog(ERROR is already so marked; what's with gcc?*/
242 case COERCION_PATH_RELABELTYPE:
243 /*
244 * Binary compatible type. No need for a special coercer.
245 * Unless ... it's a domain ....
246 */
247 if ( ! IsBinaryCoercible(fromOid, toOid) && DomainHasConstraints(toOid))
248 elog(WARNING, "disregarding domain constraints of (regtype) %d",
249 toOid);
250 return self;
251 case COERCION_PATH_COERCEVIAIO:
252 elog(ERROR, "COERCEVIAIO not implemented from (regtype) %d to %d",
253 fromOid, toOid);
254 pg_unreachable();
255 case COERCION_PATH_ARRAYCOERCE:
256 elog(ERROR, "ARRAYCOERCE not implemented from (regtype) %d to %d",
257 fromOid, toOid);
258 pg_unreachable();
259 case COERCION_PATH_FUNC:
260 break;
261 }
262
263 if(*map == 0)
264 *map = HashMap_create(7, GetMemoryChunkContext(self));
265
266 coercer = builder(self, other, funcId);
267 HashMap_putByOid(*map, other->typeId, coercer);
268 return coercer;
269}

367 if(typeClass->javaClass == 0)
368 {
369 jclass cls;

385 cls = PgObject_getJavaClass(bp);
386 pfree(bp);
387 }
388 else
389 cls = PgObject_getJavaClass(cp);
390
391 typeClass->javaClass = JNI_newGlobalRef(cls);
392 JNI_deleteLocalRef(cls);
393 }

456Datum Type_invokeSRF(Type self, Function fn, PG_FUNCTION_ARGS)
457{
458 jobject row;
459 CallContextData* ctxData;
460 FuncCallContext* context;
461 MemoryContext currCtx;
462
463 /* stuff done only on the first call of the function
464 */
465 if(SRF_IS_FIRSTCALL())
466 {
467 jobject tmp;
468
469 /* create a function context for cross-call persistence
470 */
471 context = SRF_FIRSTCALL_INIT();
472
473 /*
474 * Before creating the rowProducer (and rowCollector, if applicable),
475 * switch to the SRF_FIRSTCALL_INIT-created multi_call_memory_ctx that
476 * is not reset between calls. The motivation seems clear enough (allow
477 * the first-call initialization to allocate things in a context that
478 * will last through the sequence), though it is not clear whether
479 * anything in existing PL/Java code in fact does so (other than our
480 * allocation of ctxData below, which could perhaps just be a direct
481 * MemoryContextAllocZero).
482 */
483 currCtx = MemoryContextSwitchTo(context->multi_call_memory_ctx);
484
485 /* Call the declared Java function. It returns an instance
486 * that can produce the rows.
487 */
488 tmp = pljava_Function_refInvoke(fn);
489 if(tmp == 0)
490 {
491 Invocation_assertDisconnect();
492 MemoryContextSwitchTo(currCtx);
493 fcinfo->isnull = true;
494 SRF_RETURN_DONE(context);
495 }
496
497 ctxData = (CallContextData*)palloc0(sizeof(CallContextData));
498 context->user_fctx = ctxData;
499
500 ctxData->elemType = self;
501 ctxData->fn = fn;
502 ctxData->rowProducer = JNI_newGlobalRef(tmp);
503 JNI_deleteLocalRef(tmp);
504
505 /* Some row producers will need a writable result set in order
506 * to produce the row. If one is needed, it's created here.
507 */
508 tmp = Type_getSRFCollector(self, fcinfo);
509 if(tmp != 0)
510 {
511 ctxData->rowCollector = JNI_newGlobalRef(tmp);
512 JNI_deleteLocalRef(tmp);
513 }
514
515 stashCallContext(ctxData);
516
517 /* Register callback to be called when the function ends
518 */
519 RegisterExprContextCallback(
520 ((ReturnSetInfo*)fcinfo->resultinfo)->econtext,
521 _endOfSetCB, PointerGetDatum(ctxData));
522
523 /*
524 * Switch back to the context on entry, which by caller arrangement is
525 * one that gets reset between calls. Thus here at the conclusion of the
526 * first-call initialization, the context invariant below is satisfied.
527 */
528 MemoryContextSwitchTo(currCtx);
529 }
530
531 /*
532 * Invariant: whether this is the first call and the SRF_IS_FIRSTCALL block
533 * above just completed, or this is a subsequent call, at this point, the
534 * memory context is the per-row one supplied by the executor (which gets
535 * reset between calls).
536 */
537
538 context = SRF_PERCALL_SETUP();
539 ctxData = (CallContextData*)context->user_fctx;
540 currCtx = CurrentMemoryContext; /* save executor's per-row context */
541 currentInvocation->hasConnected = ctxData->hasConnected;
542 currentInvocation->invocation = ctxData->invocation;
543
544 if(JNI_TRUE == pljava_Function_vpcInvoke(ctxData->fn,
545 ctxData->rowProducer, ctxData->rowCollector, (jlong)context->call_cntr,
546 JNI_FALSE, &row))
547 {
548 Datum result = Type_datumFromSRF(self, row, ctxData->rowCollector);
549 JNI_deleteLocalRef(row);
550 stashCallContext(ctxData);
551 currentInvocation->hasConnected = false;
552 currentInvocation->invocation = 0;
553 MemoryContextSwitchTo(currCtx);
554 SRF_RETURN_NEXT(context, result);
555 }
556
557 stashCallContext(ctxData);
558 currentInvocation->hasConnected = false;
559 currentInvocation->invocation = 0;
560 MemoryContextSwitchTo(currCtx);
561
562 /* Unregister this callback and call it manually. We do this because
563 * otherwise it will be called when the backend is in progress of
564 * cleaning up Portals. If we close cursors (i.e. drop portals) in
565 * the close, then that mechanism fails since attempts are made to
566 * delete portals more then once.
567 */
568 UnregisterExprContextCallback(
569 ((ReturnSetInfo*)fcinfo->resultinfo)->econtext,
570 _endOfSetCB,
571 PointerGetDatum(ctxData));
572
573 _closeIteration(ctxData);
574
575 /* This is the end of the set.
576 */
577 SRF_RETURN_DONE(context);
578}

585Type Type_fromJavaType(Oid typeId, const char* javaTypeName)
586{
587 /*
588 * Do an initial lookup with InvalidOid as the oid part of the key. Multiple
589 * entries for the same Java name and distinct oids are not anticipated
590 * except for arrays.
591 */
592 CacheEntry ce = (CacheEntry)HashMap_getByStringOid(
593 s_obtainerByJavaName, javaTypeName, InvalidOid);
594
595 /*
596 * If no entry was found using InvalidOid and a valid typeId is provided
597 * and the wanted Java type is an array, repeat the lookup using the typeId.
598 */
599 if ( NULL == ce && InvalidOid != typeId
600 && NULL != strchr(javaTypeName, ']') )
601 ce = (CacheEntry)HashMap_getByStringOid(
602 s_obtainerByJavaName, javaTypeName, typeId);
603
604 if(ce == 0)
605 {
606 size_t jtlen = strlen(javaTypeName) - 2;
607 if(jtlen > 0 && strcmp("[]", javaTypeName + jtlen) == 0)
608 {
609 Type type;
610 char* elemName = palloc(jtlen+1);
611 memcpy(elemName, javaTypeName, jtlen);
612 elemName[jtlen] = 0;
613 type = Type_getArrayType(Type_fromJavaType(InvalidOid, elemName), typeId);
614 pfree(elemName);
615 return type;
616 }
617 ereport(ERROR, (
618 errcode(ERRCODE_CANNOT_COERCE),
619 errmsg("No java type mapping installed for \"%s\"", javaTypeName)));
620 }
621
622 return ce->type == 0
623 ? ce->obtainer(typeId == InvalidOid ? ce->typeId : typeId)
624 : ce->type;
625}

637/*
638 * Return NULL unless typeId represents a MappedUDT as found in the typeMap,
639 * in which case return a freshly-registered UDT Type.
640 *
641 * A MappedUDT's supporting functions don't have SQL declarations, from which
642 * an ordinary function's PLPrincipal and initiating class loader would be
643 * determined, so when obtaining the support function handles below, NULL will
644 * be passed as the language name, indicating that information isn't available,
645 * and won't be baked into the handles.
646 *
647 * A MappedUDT only has the two support functions readSQL and writeSQL.
648 * The I/O support functions parse and toString are only for a BaseUDT, so
649 * they do not need to be looked up here.
650 *
651 * The typeStruct argument supplies the type's name and namespace to
652 * UDT_registerUDT, as well as the by-value, length, and alignment common to
653 * any registered Type.
654 *
655 * A complication, though: in principle, this is a function on two variables,
656 * typeId and typeMap. (The typeStruct is functionally dependent on typeId.)
657 * But registration of the first one to be encountered will enter it in caches
658 * that depend only on the typeId (or Java class name, for the other direction)
659 * from that point on. This is longstanding PL/Java behavior, but XXX.
660 */
661static inline Type
662checkTypeMappedUDT(Oid typeId, jobject typeMap, Form_pg_type typeStruct)
663{
664 jobject joid;
665 jclass typeClass;
666 Type type;
667 jobject readMH;
668 jobject writeMH;
669 TupleDesc tupleDesc;
670 bool hasTupleDesc;
671
672 if ( NULL == typeMap )
673 return NULL;
674
675 joid = Oid_create(typeId);
676 typeClass = (jclass)JNI_callObjectMethod(typeMap, s_Map_get, joid);
677 JNI_deleteLocalRef(joid);
678
679 if ( NULL == typeClass )
680 return NULL;
681
682 if ( -2 == typeStruct->typlen )
683 {
684 JNI_deleteLocalRef(typeClass);
685 ereport(ERROR, (
686 errcode(ERRCODE_FEATURE_NOT_SUPPORTED),
687 errmsg(
688 "type mapping in PL/Java for %s with NUL-terminated(-2) "
689 "storage not supported",
690 format_type_be_qualified(typeId))
691 ));
692 }
693
694 readMH = pljava_Function_udtReadHandle( typeClass, NULL, true);
695 writeMH = pljava_Function_udtWriteHandle(typeClass, NULL, true);
696
697 tupleDesc = lookup_rowtype_tupdesc_noerror(typeId, -1, true);
698 hasTupleDesc = NULL != tupleDesc;
699 if ( hasTupleDesc )
700 ReleaseTupleDesc(tupleDesc);
701
702 type = (Type)UDT_registerUDT(
703 typeClass, typeId, typeStruct, hasTupleDesc, false,
704 NULL, readMH, writeMH, NULL);
705 /*
706 * UDT_registerUDT calls JNI_deleteLocalRef on readMH and writeMH.
707 */
708
709 JNI_deleteLocalRef(typeClass);
710 return type;
711}

713Type Type_fromOid(Oid typeId, jobject typeMap)
714{
715 CacheEntry ce;
716 HeapTuple typeTup;
717 Form_pg_type typeStruct;
718 Type type = Type_fromOidCache(typeId);
719
720 if ( NULL != type )
721 return type;
722
723 typeTup = PgObject_getValidTuple(TYPEOID, typeId, "type");
724 typeStruct = (Form_pg_type)GETSTRUCT(typeTup);
725
726 if(typeStruct->typelem != 0 && typeStruct->typlen == -1)
727 {
728 type = Type_getArrayType(
729 Type_fromOid(typeStruct->typelem, typeMap), typeId);

731 }

733 /* For some reason, the anyarray is *not* an array with anyelement as the
734 * element type. We'd like to see it that way though.
735 * XXX would we, or does that mistake something intended in PostgreSQL?
736 */
737 if(typeId == ANYARRAYOID)
738 {
739 type = Type_getArrayType(Type_fromOid(ANYELEMENTOID, typeMap), typeId);
740 goto finally;
741 }
742
743 if(typeStruct->typbasetype != 0)
744 {
745 /* Domain type, recurse using the base type (which in turn may
746 * also be a domain)
747 */
748 type = Type_fromOid(typeStruct->typbasetype, typeMap);
749 goto finally;
750 }
751
752 /*
753 * Perhaps we have found a MappedUDT. If so, this check will register and
754 * return it.
755 */
756 type = checkTypeMappedUDT(typeId, typeMap, typeStruct);
757 if ( NULL != type )
758 goto finally;
759
760 /* Composite and record types will not have a TypeObtainer registered
761 */
762 if(typeStruct->typtype == 'c'
763 || (typeStruct->typtype == 'p' && typeId == RECORDOID))
764 {
765 type = Composite_obtain(typeId);
766 goto finally;
767 }
768
769 ce = (CacheEntry)HashMap_getByOid(s_obtainerByOid, typeId);
770 if ( NULL == ce )
771 {
772 /*
773 * Perhaps we have found a BaseUDT. If so, this check will register and
774 * return it.
775 */
776 type = Function_checkTypeBaseUDT(typeId, typeStruct);
777 if ( NULL != type )
778 goto finally;
779 /*
780 * Default to String and standard textin/textout coercion.
781 * Note: if the AS spec includes a Java signature, and the corresponding
782 * Java type is not String, that will trigger a call to
783 * Type_fromJavaType to see if a mapping is registered that way. If not,
784 * *that* function reports 'No java type mapping installed for "%s"'.
785 */
786 type = String_obtain(typeId);
787 }
788 else
789 {
790 type = ce->type;
791 if(type == 0)
792 type = ce->obtainer(typeId);
793 }
794
795finally:
796 ReleaseSysCache(typeTup);
797 Type_cacheByOid(typeId, type);
798 return type;
799}
